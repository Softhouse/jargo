015package se.softhouse.common.guavaextensions;
016
017import static com.google.common.base.Preconditions.checkArgument;
018import static com.google.common.base.Preconditions.checkNotNull;
019import static com.google.common.collect.Lists.newArrayListWithCapacity;
020
021import java.util.List;
022
023import javax.annotation.concurrent.Immutable;
024
025import com.google.common.base.Supplier;
026import com.google.common.base.Suppliers;
027
028/**
029 * Additional implementations of the {@link Supplier} interface, as a complement of
030 * {@link Suppliers}
031 */
032@Immutable
033public final class Suppliers2
034{
035        private Suppliers2()
036        {
037        }
038
039        /**
040         * Creates a {@link Supplier} that supplies {@code elementsToSupply} number of elements from
041         * {@code elementSupplier}
042         * 
043         * @throws IllegalArgumentException if {@code elementsToSupply} is less than zero
044         */
045        public static <T> Supplier<List<T>> ofRepeatedElements(Supplier<? extends T> elementSupplier, int elementsToSupply)
046        {
047                checkNotNull(elementSupplier);
048                checkArgument(elementsToSupply >= 0, "elementsToSupply may not be negative");
049                return new ListSupplier<T>(elementSupplier, elementsToSupply);
050        }
051
052        private static final class ListSupplier<T> implements Supplier<List<T>>
053        {
054                private final Supplier<? extends T> elementSupplier;
055                private final int elementsToSupply;
056
057                private ListSupplier(Supplier<? extends T> elementSupplier, final int elementsToSupply)
058                {
059                        this.elementSupplier = elementSupplier;
060                        this.elementsToSupply = elementsToSupply;
061                }
062
063                @Override
064                public List<T> get()
065                {
066                        List<T> result = newArrayListWithCapacity(elementsToSupply);
067                        for(int i = 0; i < elementsToSupply; i++)
068                        {
069                                result.add(elementSupplier.get());
070                        }
071                        return result;
072                }
073        }
074
075        /**
076         * Returns true if {@code Supplier} is likely to supply values very fast
077         */
078        public static boolean isSuppliedAlready(Supplier<?> supplier)
079        {
080                if(supplier.getClass().equals(OF_INSTANCE))
081                        return true;
082                else if(supplier instanceof ListSupplier<?>)
083                {
084                        ListSupplier<?> listSupplier = (ListSupplier<?>) supplier;
085                        if(listSupplier.elementSupplier.getClass().equals(OF_INSTANCE))
086                                return true;
087                }
088                return false;
089        }
090
091        private static final Class<?> OF_INSTANCE = Suppliers.ofInstance(null).getClass();
092}
