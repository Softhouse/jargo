015    package se.softhouse.common.guavaextensions;
016    
017    import static com.google.common.base.Preconditions.checkNotNull;
018    import static com.google.common.base.Predicates.alwaysTrue;
019    
020    import java.util.List;
021    
022    import javax.annotation.Nonnull;
023    
024    import se.softhouse.common.strings.Describables;
025    
026    import com.google.common.base.Predicate;
027    import com.google.common.base.Predicates;
028    
029    /**
030     * Additional implementations of the {@link Predicate} interface, as a complement to
031     * {@link Predicates}
032     */
033    public final class Predicates2
034    {
035            private Predicates2()
036            {
037            }
038    
039            /**
040             * Returns a predicate that throws an {@link IllegalArgumentException} if any element in a list
041             * doesn't return true for {@code elementLimiter.apply(Object)}. Useful for input validation
042             * where you already have a {@link Predicate} for the element type and want to check a list of
043             * such elements.
044             * Without {@link #listPredicate(Predicate)} it would look something like:
045             * 
046             * <pre class="prettyprint">
047             * <code class="language-java">
048             * List&lt;Integer&gt; numbers = Arrays.asList(1, -1, -2);
049             * Optional&lt;Integer&gt; invalidNumber = Iterables.tryFind(numbers, Predicates.not(ABOVE_ZERO));
050             * if(invalidNumber.isPresent())
051             *   throw new IllegalArgumentException("'" + invalidNumber.get() + "' is not " + ABOVE_ZERO);
052             * </code>
053             * </pre>
054             * 
055             * With {@link #listPredicate(Predicate)}:
056             * 
057             * <pre class="prettyprint">
058             * <code class="language-java">
059             * Predicates2.listPredicate(ABOVE_ZERO).apply(numbers);
060             * </code>
061             * </pre>
062             * 
063             * As this breaks the general contract of {@link Predicate#apply(Object)} it should be
064             * considered useful for validating arguments only.
065             * The detail message only mentions the first element that didn't return true.
066             */
067            public static <E> Predicate<List<? extends E>> listPredicate(Predicate<E> elementLimiter)
068            {
069                    checkNotNull(elementLimiter);
070                    if(elementLimiter == alwaysTrue())
071                            return alwaysTrue();
072                    return new ListPredicate<E>(elementLimiter);
073            }
074    
075            private static final class ListPredicate<E> implements Predicate<List<? extends E>>
076            {
077                    private final Predicate<E> elementLimiter;
078    
079                    private ListPredicate(Predicate<E> elementLimiter)
080                    {
081                            this.elementLimiter = elementLimiter;
082                    }
083    
084                    @Override
085                    public boolean apply(@Nonnull List<? extends E> values)
086                    {
087                            for(E value : values)
088                            {
089                                    if(!elementLimiter.apply(value))
090                                            throw Describables.illegalArgument(Describables.format("'%s' is not %s", value, elementLimiter));
091                            }
092                            return true;
093                    }
094    
095                    @Override
096                    public String toString()
097                    {
098                            return elementLimiter.toString();
099                    }
100            }
101    
102            /**
103             * Works just like {@link Predicates#and(Predicate, Predicate)} except that if {@code first} is
104             * {@link Predicates#alwaysTrue()} {@code second} is returned directly (or vice versa). This has
105             * the potential to make {@link Predicate#toString()} look a bit nicer for the resulting
106             * {@link Predicate}.
107             */
108            public static <T> Predicate<T> and(Predicate<T> first, Predicate<T> second)
109            {
110                    if(first == alwaysTrue())
111                            return checkNotNull(second);
112                    else if(second == alwaysTrue())
113                            return checkNotNull(first);
114                    return Predicates.and(first, second);
115            }
116    }
