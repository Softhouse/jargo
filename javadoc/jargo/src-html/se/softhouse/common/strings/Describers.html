049        /**
050         * Always describes any value of type {@code T} with the given {@code constant}. For instance,
051         * if you have implemented a time related parser and don't want different times in the
052         * usage depending on when the usage is printed you could pass in "The current time", then this
053         * describer would describe any time object with "The current time".
054         */
055        @Nonnull
056        @CheckReturnValue
057        public static <T> Describer<T> withConstantString(final String constant)
058        {
059                return new ConstantStringDescriber<T>(constant);
060        }
061
062        private static final class ConstantStringDescriber<T> implements Describer<T>
063        {
064                private final String constant;
065
066                private ConstantStringDescriber(final String constant)
067                {
068                        this.constant = checkNotNull(constant);
069                }
070
071                @Override
072                public String describe(T value, Locale inLocale)
073                {
074                        return constant;
075                }
076        }

078        /**
079         * Calls {@link String#valueOf(Object)} for input values. As this goes
080         * against the very purpose of the {@link Describer} interface it may seem
081         * odd but this {@link Describer} works really well as a null object, it can
082         * also act as a functor for calling toString.
083         */
084        @Nonnull
085        @CheckReturnValue
086        public static <T> Describer<T> toStringDescriber()
087        {
088                @SuppressWarnings("unchecked")
089                Describer<T> toStringDescriber = (Describer<T>) ToStringDescriber.INSTANCE;
090                return toStringDescriber;
091        }

099                }

225        /**
226         * Describes what a key=value in a {@link Map} means by fetching a
227         * description from {@code descriptions} for each key in a given map. So {@code descriptions}
228         * need to have values (descriptions) for all keys in any given map,
229         * otherwise a {@link NullPointerException} is thrown. For example:
230         * 
231         * <pre class="prettyprint">
232         * <code class="language-java">
233         * Map&lt;String, Integer&gt; defaults = newLinkedHashMap();
234         * defaults.put("population", 42);
235         * defaults.put("hello", 1);
236         * 
237         * Map&lt;String, String&gt; descriptions = newLinkedHashMap();
238         * descriptions.put("population", "The number of citizens in the world");
239         * descriptions.put("hello", "The number of times to say hello");
240         * Describer&lt;Map&lt;String, Integer&gt;&gt; d = mapDescriber(descriptions);
241         * 
242         * String describedMap = d.describe(defaults);
243         * </code>
244         * </pre>
245         * 
246         * would return:
247         * 
248         * <pre>
249         * <code>
250         * population=42
251         *      The number of citizens in the world
252         * hello=1
253         *      The number of times to say hello
254         * </code>
255         * </pre>
256         * 
257         * You could even implement a
258         * {@code addProperty(String key, int defaultValue, String description)} method to enforce the
259         * use of descriptions at compile-time.
260         */
261        @Nonnull
262        @CheckReturnValue
263        public static <K, V> Describer<Map<K, V>> mapDescriber(Map<K, String> descriptions)
264        {
265                return new MapDescription<K, V>(ImmutableMap.copyOf(descriptions), Describers.<K>toStringDescriber());
266        }
267
268        /**
269         * Works like {@link #mapDescriber(Map)} but it describes keys in any given {@link Map} with
270         * {@code keyDescriber} instead of with {@link #toString()}
271         * 
272         * @param descriptions a map with strings describing what each key in it means
273         * @param keyDescriber {@link Describer} used to transform keys into {@link String}s with
274         * @return a {@link Describer} that can describe a map of the type {@code Map<K, V>} and where
275         *         each key must have a corresponding description in {@code descriptions}
276         */
277        public static <K, V> Describer<Map<K, V>> mapDescriber(Map<K, String> descriptions, Describer<K> keyDescriber)
278        {
279                return new MapDescription<K, V>(ImmutableMap.copyOf(descriptions), checkNotNull(keyDescriber));
280        }
281
282        private static final class MapDescription<K, V> implements Describer<Map<K, V>>
283        {
284                private final Map<K, String> descriptions;
285                private final Describer<K> keyDescriber;
286
287                private MapDescription(Map<K, String> descriptions, Describer<K> keyDescriber)
288                {
289                        this.descriptions = descriptions;
290                        this.keyDescriber = keyDescriber;
291                }
292
293                @Override
294                public String describe(Map<K, V> values, Locale inLocale)
295                {
296                        StringBuilder result = new StringBuilder();
297                        for(Entry<K, V> entry : values.entrySet())
298                        {
299                                K key = entry.getKey();
300                                result.append(keyDescriber.describe(key, inLocale));
301                                result.append("=");
302                                result.append(entry.getValue());
303                                String descriptionForEntry = descriptions.get(key);
304                                checkNotNull(descriptionForEntry, "Undescribed key: %s", key);
305                                result.append(NEWLINE).append(" ").append(descriptionForEntry).append(NEWLINE);
306                        }
307                        return result.toString();
308                }
309        }

361        private static final class MapDescriber<K, V> implements Describer<Map<K, V>>
362        {
363                private final Describer<V> valueDescriber;
364                private final Describer<K> keyDescriber;
365                private final String valueSeparator;
366
367                private MapDescriber(Describer<K> keyDescriber, Describer<V> valueDescriber, String valueSeparator)
368                {
369                        this.keyDescriber = checkNotNull(keyDescriber);
370                        this.valueDescriber = checkNotNull(valueDescriber);
371                        this.valueSeparator = checkNotNull(valueSeparator);
372                }
373
374                @Override
375                public String describe(Map<K, V> values, Locale inLocale)
376                {
377                        if(values == null)
378                                return NULL;
379                        Iterator<Entry<K, V>> iterator = values.entrySet().iterator();
380                        if(!iterator.hasNext())
381                                return "Empty map";
382
383                        StringBuilder firstKeyValue = new StringBuilder();
384                        describeEntry(iterator.next(), inLocale, firstKeyValue);
385
386                        StringBuilder result = StringBuilders.withExpectedSize(firstKeyValue.length() * values.size());
387                        result.append(firstKeyValue);
388
389                        while(iterator.hasNext())
390                        {
391                                result.append(NEWLINE);
392                                describeEntry(iterator.next(), inLocale, result);
393                        }
394                        return result.toString();
395                }
396
397                private void describeEntry(Entry<K, V> entry, Locale inLocale, StringBuilder output)
398                {
399                        output.append(keyDescriber.describe(entry.getKey(), inLocale));
400                        output.append(valueSeparator);
401                        output.append(valueDescriber.describe(entry.getValue(), inLocale));
402                }
403        }

405        /**
406         * <pre>
407         * Exposes a {@link Describer} as a Guava {@link Function}.
408         * <b>Note:</b>This method may be removed in the future if Guava is removed as a dependency.
409         * 
410         * @param describer the describer to convert to a {@link Function}
411         * @return a {@link Function} that uses {@link Describer#describe(Object, Locale) describe(input, Locale.getDefault()}.
412         * </pre>
413         */
414        @Beta
415        @Nonnull
416        @CheckReturnValue
417        public static <T> Function<? super T, String> asFunction(final Describer<T> describer)
418        {
419                checkNotNull(describer);
420                return new Function<T, String>(){
421                        @Override
422                        public String apply(@Nonnull T input)
423                        {
424                                return describer.describe(input, Locale.getDefault());
425                        }
426                };
427        }

486         * Describes values in lists with {@code valueDescriber} and separates

488         * is described with the string "Empty List".
489         */
490        @Nonnull
491        @CheckReturnValue
492        public static <T> Describer<List<? extends T>> listDescriber(Describer<T> valueDescriber, String valueSeparator)
493        {
494                return new ListDescriber<T>(valueDescriber, valueSeparator);
495        }
496
497        private static final class ListDescriber<T> implements Describer<List<? extends T>>
498        {
499                private final Describer<T> valueDescriber;
500                private final String valueSeparator;
501
502                ListDescriber(Describer<T> valueDescriber, String valueSeparator)
503                {
504                        this.valueDescriber = checkNotNull(valueDescriber);
505                        this.valueSeparator = checkNotNull(valueSeparator);
506                }
507
508                @Override
509                public String describe(List<? extends T> value, Locale inLocale)
510                {
511                        if(value == null)
512                                return NULL;
513                        if(value.isEmpty())
514                                return "Empty list";
515
516                        Iterator<? extends T> values = value.iterator();
517                        String firstValue = valueDescriber.describe(values.next(), inLocale);
518
519                        StringBuilder sb = StringBuilders.withExpectedSize(value.size() * firstValue.length());
520
521                        sb.append(firstValue);
522                        while(values.hasNext())
523                        {
524                                sb.append(valueSeparator).append(valueDescriber.describe(values.next(), inLocale));
525                        }
526                        return sb.toString();
527                }
528        }
