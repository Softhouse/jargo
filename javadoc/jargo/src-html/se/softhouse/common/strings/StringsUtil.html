083            /**
084             * <pre>
085             * Returns the {@link String} in {@code validOptions} that {@code input} has the shortest
086             * <a href="http://en.wikipedia.org/wiki/Levenshtein_distance">levenshtein distance</a> to.
087             * 
088             * For example when given "stats" as input and "status", "help",
089             * "action" as validOptions, "status" is returned.
090             * 
091             * Current performance characteristics:
092             * n = length of {@code input}
093             * m = average string length of the strings in {@code validOptions}
094             * s = amount of validOptions
095             * 
096             * complexity = n * m * s = O(n<sup>3</sup>)
097             * 
098             * So try to limit the number of valid options...
099             * 
100             * @throws IllegalArgumentException if {@code validOptions} is empty
101             * </pre>
102             * 
103             * @see #closestMatches(String, Iterable, int)
104             */
105            @Nonnull
106            @CheckReturnValue
107            public static String closestMatch(final String input, final Iterable<String> validOptions)
108            {
109                    checkNotNull(input);
110                    checkArgument(!isEmpty(validOptions), "No valid options to match the input against");
111    
112                    int shortestDistance = Integer.MAX_VALUE;
113                    String bestGuess = null;
114                    for(String validOption : validOptions)
115                    {
116                            int distance = levenshteinDistance(input, validOption);
117                            if(distance < shortestDistance)
118                            {
119                                    shortestDistance = distance;
120                                    bestGuess = validOption;
121                            }
122                    }
123                    return bestGuess;
124            }

126            /**
127             * <pre>
128             * Returns a sorted {@link List} where the first entry is the {@link String} in {@code validOptions} that's closest in terms of
129             * <a href="http://en.wikipedia.org/wiki/Levenshtein_distance">levenshtein distance</a> to {@code input}, or an empty list if no options within distance can be found.
130             * 
131             * For example when given "stats" as input and "status", "staging",
132             * "stage" as validOptions, and 4 as maximumDistance, "status", "stage", "staging" is returned.
133             * 
134             * Only values with a distance less than or equal to {@code maximumDistance} will be included in the result.
135             * 
136             * The returned list is <i>modifiable</i>.
137             * 
138             * </pre>
139             */
140            @Nonnull
141            @CheckReturnValue
142            public static List<String> closestMatches(final String input, final Iterable<String> validOptions, int maximumDistance)
143            {
144                    checkNotNull(input);
145                    if(isEmpty(validOptions))
146                            return Collections.emptyList();
147    
148                    List<CloseMatch> closeMatches = Lists.newArrayList();
149                    for(String validOption : validOptions)
150                    {
151                            int distance = levenshteinDistance(input, validOption);
152                            if(distance <= maximumDistance)
153                            {
154                                    closeMatches.add(new CloseMatch(validOption, distance));
155                            }
156                    }
157                    Collections.sort(closeMatches, BY_CLOSEST_MATCH_FIRST);
158                    return Lists.transform(closeMatches, CloseMatch.GET_VALUE);
159            }
160    
161            static final class CloseMatch
162            {
163                    private final int measuredDistance;
164                    private final String value;
165    
166                    private CloseMatch(String validOption, int distance)
167                    {
168                            measuredDistance = distance;
169                            value = validOption;
170                    }
171    
172                    static final Comparator<CloseMatch> BY_CLOSEST_MATCH_FIRST = new Comparator<CloseMatch>(){
173                            @Override
174                            public int compare(CloseMatch left, CloseMatch right)
175                            {
176                                    return left.measuredDistance - right.measuredDistance;
177                            }
178                    };
179    
180                    private static final Function<CloseMatch, String> GET_VALUE = new Function<CloseMatch, String>(){
181                            @Override
182                            public String apply(@Nonnull CloseMatch input)
183                            {
184                                    return input.value;
185                            }
186                    };
187            }

189            /**
190             * Returns the <a href="http://en.wikipedia.org/wiki/Levenshtein_distance">levenshtein
191             * distance</a> between {@code left} and {@code right}.
192             * 
193             * @see #closestMatch(String, Iterable)
194             */
195            public static int levenshteinDistance(final String left, final String right)
196            {
197                    checkNotNull(left);
198                    checkNotNull(right);
199    
200                    // a "cleaner" version of the org.apache.commons-lang algorithm which in
201                    // turn was inspired by http://www.merriampark.com/ldjava.htm
202                    int leftLength = left.length();
203                    int rightLength = right.length();
204    
205                    if(leftLength == 0)
206                            return rightLength;
207                    else if(rightLength == 0)
208                            return leftLength;
209    
210                    int previousDistances[] = new int[leftLength + 1]; // 'previous' cost array, horizontally
211                    int distances[] = new int[leftLength + 1]; // cost array, horizontally
212    
213                    int leftIndex;
214                    int rightIndex;
215    
216                    char rightChar;
217    
218                    for(leftIndex = 0; leftIndex <= leftLength; leftIndex++)
219                    {
220                            previousDistances[leftIndex] = leftIndex;
221                    }
222    
223                    for(rightIndex = 1; rightIndex <= rightLength; rightIndex++)
224                    {
225                            rightChar = right.charAt(rightIndex - 1);
226                            distances[0] = rightIndex;
227    
228                            for(leftIndex = 1; leftIndex <= leftLength; leftIndex++)
229                            {
230                                    int insertionCost = distances[leftIndex - 1] + 1;
231                                    int editCost = previousDistances[leftIndex] + 1;
232                                    int deletionCost = previousDistances[leftIndex - 1];
233                                    if(left.charAt(leftIndex - 1) != rightChar)
234                                    {
235                                            deletionCost++;
236                                    }
237    
238                                    distances[leftIndex] = Ints.min(insertionCost, editCost, deletionCost);
239                            }
240    
241                            // Swap current distance counts to 'previous row' distance counts
242                            int[] temp = previousDistances;
243                            previousDistances = distances;
244                            distances = temp;
245                    }
246    
247                    // our last action in the above loop was to switch distances and
248                    // previousDistances, so
249                    // previousDistances now actually has the most recent cost counts
250                    return previousDistances[leftLength];
251            }

253            /**
254             * Returns {@code number} expressed as a position. For example 0 returns
255             * "zeroth", 1 returns "first" and so forth up to "fifth". Higher positions
256             * are described as "6th", "7th" and so on.
257             * 
258             * @throws IllegalArgumentException if {@code number} is negative
259             */
260            @Nonnull
261            @CheckReturnValue
262            public static String numberToPositionalString(int number)
263            {
264                    checkArgument(number >= 0, "Negative numbers don't have positions");
265                    switch(number)
266                    {
267                    case 0:
268                            return "zeroth";
269                    case 1:
270                            return "first";
271                    case 2:
272                            return "second";
273                    case 3:
274                            return "third";
275                    case 4:
276                            return "fourth";
277                    case 5:
278                            return "fifth";
279    
280                    }
281                    return Integer.toString(number) + "th";
282            }

284            /**
285             * Finds the {@code nth} occurrence of {@code needle} in {@code haystack}
286             * 
287             * @param nth how many occurrences of {@code needle} that should occur before the returned index
288             * @param needle the string to search for
289             * @param haystack the string to search within
290             * @return the starting index of the {@code nth} occurrence of {@code needle} within
291             *         {@code haystack}, -1 if {@code nth} occurrences couldn't be found
292             */
293            public static int indexOfNth(int nth, String needle, String haystack)
294            {
295                    checkNotNull(haystack);
296                    checkNotNull(needle);
297                    checkArgument(nth > 0, "nth must be at least 1 (was %s)", nth);
298                    int occurencesFound = 0;
299                    int index = -1;
300                    while(occurencesFound < nth)
301                    {
302                            index = haystack.indexOf(needle, index + 1);
303                            occurencesFound++;
304                            if(index == -1)
305                            {
306                                    break;
307                            }
308                    }
309                    return index;
310            }
