015package se.softhouse.common.strings;
016
017import static se.softhouse.common.strings.StringsUtil.NEWLINE;
018
019import java.text.BreakIterator;
020import java.util.Locale;
021
022/**
023 * Utilities for working with {@link StringsUtil#NEWLINE new line} in texts.
024 */
025public final class Lines
026{
027        private Lines()
028        {
029        }
030
031        /**
032         * Wraps lines where <a
033         * href="http://docs.oracle.com/javase/tutorial/i18n/text/line.html">appropriate</a> (as defined
034         * by {@code locale}).
035         * 
036         * @param value the value to separate with {@link StringsUtil#NEWLINE new lines}
037         * @param maxLineLength how long each line are allowed to be
038         */
039        public static StringBuilder wrap(CharSequence value, int maxLineLength, Locale locale)
040        {
041                return wrap(value, 0, maxLineLength, locale);
042        }
043
044        /**
045         * Wraps lines where <a
046         * href="http://docs.oracle.com/javase/tutorial/i18n/text/line.html">appropriate</a> (as defined
047         * by {@code locale}).
048         * 
049         * @param value the value to separate with {@link StringsUtil#NEWLINE new lines}
050         * @param startingIndex the index where each line starts, useful for a fixed-size table for
051         *            instance
052         * @param maxLineLength how long each line are allowed to be
053         */
054        public static StringBuilder wrap(CharSequence value, int startingIndex, int maxLineLength, Locale locale)
055        {
056                String textToSplit = value.toString();
057                StringBuilder result = new StringBuilder(textToSplit.length());
058                BreakIterator boundary = BreakIterator.getLineInstance(locale);
059                boundary.setText(textToSplit);
060                int start = boundary.first();
061                int end = boundary.next();
062                int lineLength = startingIndex;
063
064                while(end != BreakIterator.DONE)
065                {
066                        String word = textToSplit.substring(start, end);
067                        lineLength = lineLength + word.length();
068                        if(lineLength >= maxLineLength)
069                        {
070                                result.append(NEWLINE);
071                                lineLength = startingIndex;
072                        }
073                        result.append(word);
074                        start = end;
075                        end = boundary.next();
076                }
077                return result;
078        }
079}
