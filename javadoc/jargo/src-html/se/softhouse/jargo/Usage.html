049    /**
050     * Responsible for formatting usage texts for {@link CommandLineParser#usage()} and
051     * {@link Arguments#helpArgument(String, String...)}.
052     * Using it is often as simple as:
053     * 
054     * <pre class="prettyprint">
055     * <code class="language-java">
056     * System.out.print(CommandLineParser.withArguments(someArguments).usage());
057     * </code>
058     * </pre>
059     * 
060     * <pre>
061     * Sorts {@link Argument}s in the following order:
062     * <ol>
063     *   <li>{@link ArgumentBuilder#names(String...) indexed arguments} without a {@link ArgumentBuilder#variableArity() variable arity}</li>
064     *       <li>By their {@link ArgumentBuilder#names(String...) first name}  in an alphabetical order</li>
065     *       <li>The remaining {@link ArgumentBuilder#names(String...) indexed arguments} that are of {@link ArgumentBuilder#variableArity() variable arity}</li>
066     * </ol>
067     * </pre>
068     */
069    @NotThreadSafe
070    public final class Usage implements Serializable
071    {
072            private static final long serialVersionUID = 1L;
073    
074            /**
075             * Corresponds to usage for a single {@link Argument}
076             */
077            @VisibleForTesting
078            static final class Row
079            {
080                    private final StringBuilder nameColumn = new StringBuilder();
081                    private final StringBuilder descriptionColumn = new StringBuilder();
082    
083                    @Override
084                    public String toString()
085                    {
086                            return nameColumn.toString() + " " + descriptionColumn.toString();
087                    }
088            }
089    
090            private final transient Collection<Argument<?>> unfilteredArguments;
091            private transient String errorMessage = "";
092            private final transient Locale locale;
093            private final transient ProgramInformation program;
094            private final transient boolean forCommand;
095            // TODO(jontejj): try getting the correct value automatically, if not possible fall back to 80
096            private transient int columnWidth = 80;
097    
098            private transient String fromSerializedUsage = null;
099    
100            /**
101             * <pre>
102             * For:
103             * -l, --enable-logging Output debug information to standard out
104             * -p, --listen-port    The port clients should connect to.
105             * 
106             * This would be 20.
107             * </pre>
108             */
109            private transient int indexOfDescriptionColumn;
110            private transient ImmutableList<Argument<?>> argumentsToPrint;
111    
112            Usage(Collection<Argument<?>> arguments, Locale locale, ProgramInformation program, boolean forCommand)
113            {
114                    this.unfilteredArguments = arguments;
115                    this.locale = locale;
116                    this.program = program;
117                    this.forCommand = forCommand;
118            }
119    
120            private Usage(String fromSerializedUsage)
121            {
122                    // All these are unused as the usage is already constructed
123                    this(null, null, null, false);
124                    this.fromSerializedUsage = fromSerializedUsage;
125            }
126    
127            /**
128             * An optional errorMessage to print before any usage
129             */
130            Usage withErrorMessage(String message)
131            {
132                    this.errorMessage = message;
133                    return this;
134            }
135    
136            /**
137             * Returns the usage text that's suitable to print on {@link System#out}.
138             */
139            @Override
140            public String toString()
141            {
142                    return usage();
143            }
144    
145            private String usage()
146            {
147                    if(fromSerializedUsage != null)
148                            return fromSerializedUsage;
149    
150                    init();
151                    StringBuilder builder = newStringBuilder();
152    
153                    printOn(builder);
154    
155                    return builder.toString();
156            }
157    
158            /**
159             * Appends usage to {@code target}. An alternative to {@link #toString()} when the usage is very
160             * large and needs to be flushed from time to time.
161             * 
162             * @throws IOException If an I/O error occurs while appending usage
163             */
164            public void printOn(Appendable target) throws IOException
165            {
166                    if(fromSerializedUsage != null)
167                    {
168                            target.append(fromSerializedUsage);
169                    }
170                    else
171                    {
172                            init();
173                            appendUsageTo(target);
174                    }
175            }
176    
177            /**
178             * Appends usage to {@code target}, just like {@link #printOn(Appendable)} but for a
179             * {@link PrintStream} instead. The reason for this overload is that {@link Appendable} declares
180             * {@link IOException} to be thrown while {@link PrintStream} handles it internally through the
181             * use of the {@link PrintStream#checkError()} method.
182             * 
183             * @param target a {@link PrintStream} such as {@link System#out} or {@link System#err}.
184             */
185            public void printOn(PrintStream target)
186            {
187                    try
188                    {
189                            printOn((Appendable) target);
190                    }
191                    catch(IOException impossible)
192                    {
193                            throw new AssertionError(impossible);
194                    }
195            }
196    
197            /**
198             * Appends usage to {@code target}, just like {@link #printOn(Appendable)} but for a
199             * {@link StringBuilder} instead
200             */
201            public void printOn(StringBuilder target)
202            {
203                    try
204                    {
205                            printOn((Appendable) target);
206                    }
207                    catch(IOException impossible)
208                    {
209                            throw new AssertionError(impossible);
210                    }
211            }
212    
213            private void appendUsageTo(Appendable builder) throws IOException
214            {
215                    builder.append(errorMessage);
216                    builder.append(header());
217    
218                    for(Argument<?> arg : argumentsToPrint)
219                    {
220                            Row forArgument = usageForArgument(arg);
221                            appendRowTo(forArgument, builder);
222                    }
223            }
224    
225            private String header()
226            {
227                    if(forCommand) // Commands get their header from their meta description
228                            return hasArguments() ? NEWLINE : "";
229    
230                    String mainUsage = UsageTexts.USAGE_HEADER + program.programName();
231    
232                    if(hasArguments())
233                    {
234                            mainUsage += UsageTexts.ARGUMENT_INDICATOR;
235                    }
236    
237                    mainUsage += NEWLINE + wordWrap(program.programDescription(), 0, columnWidth);
238    
239                    if(hasArguments())
240                    {
241                            mainUsage += NEWLINE + UsageTexts.ARGUMENT_HEADER + ":" + NEWLINE;
242                    }
243    
244                    return mainUsage;
245            }
246    
247            private static final int SPACES_BETWEEN_COLUMNS = 4;
248    
249            private void init()
250            {
251                    if(argumentsToPrint == null)
252                    {
253                            Collection<Argument<?>> visibleArguments = filter(unfilteredArguments, IS_VISIBLE);
254                            this.argumentsToPrint = copyOf(sortedArguments(visibleArguments));
255                            this.indexOfDescriptionColumn = determineLongestNameColumn() + SPACES_BETWEEN_COLUMNS;
256                    }
257            }
258    
259            private Iterable<Argument<?>> sortedArguments(Collection<Argument<?>> arguments)
260            {
261                    Collection<Argument<?>> indexedArguments = filter(arguments, IS_INDEXED);
262                    Iterable<Argument<?>> indexedWithoutVariableArity = filter(indexedArguments, not(IS_OF_VARIABLE_ARITY));
263                    Iterable<Argument<?>> indexedWithVariableArity = filter(indexedArguments, IS_OF_VARIABLE_ARITY);
264    
265                    List<Argument<?>> sortedArgumentsByName = newArrayList(filter(arguments, not(IS_INDEXED)));
266                    Collections.sort(sortedArgumentsByName, Argument.NAME_COMPARATOR);
267    
268                    return Iterables.concat(indexedWithoutVariableArity, sortedArgumentsByName, indexedWithVariableArity);
269            }
270    
271            private int determineLongestNameColumn()
272            {
273                    int longestNameSoFar = 0;
274                    for(Argument<?> arg : argumentsToPrint)
275                    {
276                            longestNameSoFar = max(longestNameSoFar, lengthOfNameColumn(arg));
277                    }
278                    return Math.min(longestNameSoFar, maxNameColumnWidth());
279            }
280    
281            private int lengthOfNameColumn(final Argument<?> argument)
282            {
283                    int namesLength = 0;
284    
285                    for(String name : argument.names())
286                    {
287                            namesLength += name.length();
288                    }
289                    int separatorLength = max(0, UsageTexts.NAME_SEPARATOR.length() * (argument.names().size() - 1));
290    
291                    int metaLength = argument.metaDescriptionInLeftColumn().length();
292    
293                    return namesLength + separatorLength + metaLength;
294            }
295    
296            /**
297             * A minimum of 1 third must be available to print descriptions on
298             */
299            private int maxNameColumnWidth()
300            {
301                    return columnWidth / 3 * 2;
302            }
303    
304            private StringBuilder newStringBuilder()
305            {
306                    // Two lines for each argument
307                    return StringBuilders.withExpectedSize(2 * argumentsToPrint.size() * columnWidth);
308            }
309    
310            private boolean hasArguments()
311            {
312                    return !argumentsToPrint.isEmpty();
313            }
314    
315            private static final Joiner NAME_JOINER = Joiner.on(UsageTexts.NAME_SEPARATOR);
316    
317            /**
318             * <pre>
319             *      -foo   Foo something [Required]
320             *              Valid values: 1 to 5
321             *        -bar   Bar something
322             *              Default: 0
323             * </pre>
324             */
325            private Row usageForArgument(final Argument<?> arg)
326            {
327                    Row row = new Row();
328    
329                    // Left column: name <meta>
330                    NAME_JOINER.appendTo(row.nameColumn, arg.names());
331                    row.nameColumn.append(arg.metaDescriptionInLeftColumn());
332    
333                    // Right column: description of what the argument means [indicators]
334                    // <meta>: description of valid values
335                    // Default: default value
336                    String description = arg.description();
337                    if(!description.isEmpty())
338                    {
339                            row.descriptionColumn.append(wordWrap(description, indexOfDescriptionColumn, columnWidth));
340                            addIndicators(arg, row.descriptionColumn);
341                            row.descriptionColumn.append(NEWLINE);
342                            valueExplanation(arg, row.descriptionColumn);
343                    }
344                    else
345                    {
346                            valueExplanation(arg, row.descriptionColumn);
347                            addIndicators(arg, row.descriptionColumn);
348                    }
349                    return row;
350            }
351    
352            private <T> void addIndicators(final Argument<T> arg, StringBuilder target)
353            {
354                    if(arg.isRequired())
355                    {
356                            target.append(UsageTexts.REQUIRED);
357                    }
358                    if(arg.isAllowedToRepeat())
359                    {
360                            target.append(UsageTexts.ALLOWS_REPETITIONS);
361                    }
362            }
363    
364            private <T> void valueExplanation(final Argument<T> arg, StringBuilder target)
365            {
366                    String validValuesDescription = arg.descriptionOfValidValues(locale);
367                    if(!validValuesDescription.isEmpty())
368                    {
369                            String meta = arg.metaDescriptionInRightColumn();
370                            target.append(meta + ": ").append(validValuesDescription);
371                    }
372                    if(arg.isRequired())
373                            return;
374    
375                    String descriptionOfDefaultValue = arg.defaultValueDescription(locale);
376                    if(descriptionOfDefaultValue != null)
377                    {
378                            if(!validValuesDescription.isEmpty())
379                            {
380                                    target.append(NEWLINE);
381                            }
382                            String spaces = spaces(UsageTexts.DEFAULT_VALUE_START.length());
383                            descriptionOfDefaultValue = descriptionOfDefaultValue.replace(NEWLINE, NEWLINE + spaces);
384                            target.append(UsageTexts.DEFAULT_VALUE_START).append(descriptionOfDefaultValue);
385                    }
386            }
387    
388            private StringBuilder wordWrap(String value, int startingLength, int maxLength)
389            {
390                    // TODO(jontejj): move into StringsUtil?
391                    StringBuilder result = new StringBuilder(value.length());
392                    BreakIterator boundary = BreakIterator.getLineInstance(locale);
393                    boundary.setText(value);
394                    int start = boundary.first();
395                    int end = boundary.next();
396                    int lineLength = startingLength;
397    
398                    while(end != BreakIterator.DONE)
399                    {
400                            String word = value.substring(start, end);
401                            lineLength = lineLength + word.length();
402                            if(lineLength >= maxLength)
403                            {
404                                    result.append(NEWLINE);
405                                    lineLength = startingLength;
406                            }
407                            result.append(word);
408                            start = end;
409                            end = boundary.next();
410                    }
411                    return result;
412            }
413    
414            private void appendRowTo(Row row, Appendable target) throws IOException
415            {
416                    StringBuilder nameColumn = wordWrap(row.nameColumn.toString(), 0, indexOfDescriptionColumn);
417                    String descriptionColumn = row.descriptionColumn.toString();
418                    Iterable<String> nameLines = Splitter.on(NEWLINE).split(nameColumn);
419                    Iterator<String> descriptionLines = Splitter.on(NEWLINE).split(descriptionColumn).iterator();
420                    for(String nameLine : nameLines)
421                    {
422                            target.append(nameLine);
423                            if(descriptionLines.hasNext())
424                            {
425                                    int lengthOfNameColumn = nameLine.length();
426                                    target.append(spaces(indexOfDescriptionColumn - lengthOfNameColumn));
427                                    target.append(descriptionLines.next());
428                            }
429                            target.append(NEWLINE);
430                    }
431                    while(descriptionLines.hasNext())
432                    {
433                            target.append(spaces(indexOfDescriptionColumn));
434                            target.append(descriptionLines.next());
435                            target.append(NEWLINE);
436                    }
437            }

439            private static final class SerializationProxy implements Serializable
440            {
441                    /**
442                     * @serial all arguments described. Constructed lazily when serialized.
443                     */
444                    private final String serializedUsage;
445    
446                    private static final long serialVersionUID = 1L;
447    
448                    private SerializationProxy(Usage usage)
449                    {
450                            // TODO(jontejj): how to support calling withConsoleWidth() after serialization? Some
451                            // kind of marker where the break iterator did something?
452                            serializedUsage = usage.usage();
453                    }
454    
455                    private Object readResolve()
456                    {
457                            return new Usage(serializedUsage);
458                    }
459            }
460    
461            private Object writeReplace()
462            {
463                    return new SerializationProxy(this);
464            }
