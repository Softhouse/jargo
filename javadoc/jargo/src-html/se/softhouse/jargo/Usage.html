017import static com.google.common.base.Predicates.not;
018import static com.google.common.collect.Collections2.filter;
019import static com.google.common.collect.ImmutableList.copyOf;
020import static com.google.common.collect.Lists.newArrayList;
021import static com.google.common.collect.Ordering.usingToString;
022import static java.lang.Math.max;
023import static se.softhouse.common.strings.StringsUtil.NEWLINE;
024import static se.softhouse.common.strings.StringsUtil.spaces;
025import static se.softhouse.jargo.Argument.IS_INDEXED;
026import static se.softhouse.jargo.Argument.IS_OF_VARIABLE_ARITY;
027import static se.softhouse.jargo.Argument.IS_VISIBLE;
028
029import java.util.Collection;
030import java.util.Collections;
031import java.util.List;
032import java.util.Locale;
033
034import javax.annotation.concurrent.NotThreadSafe;
035
036import se.softhouse.common.strings.StringBuilders;
037import se.softhouse.jargo.internal.Texts.UsageTexts;
038
039import com.google.common.base.Joiner;
040import com.google.common.collect.ImmutableList;
041import com.google.common.collect.Iterables;
042
043/**
044 * Responsible for formatting usage texts for {@link CommandLineParser#usage()}.
045 * Using it is often as simple as:
046 * 
047 * <pre class="prettyprint">
048 * <code class="language-java">
049 * System.out.print(CommandLineParser.withArguments(someArguments).usage());
050 * </code>
051 * </pre>
052 */
053@NotThreadSafe
054public final class Usage
055{
056        private static final int CHARACTERS_IN_AVERAGE_ARGUMENT_DESCRIPTION = 40;
057        private static final int SPACES_BETWEEN_COLUMNS = 4;
058        private static final Joiner NAME_JOINER = Joiner.on(UsageTexts.NAME_SEPARATOR);
059
060        private final ImmutableList<Argument<?>> argumentsToPrint;
061        private final Locale locale;
062        private final ProgramInformation program;
063
064        /**
065         * <pre>
066         * For:
067         * -l, --enable-logging Output debug information to standard out
068         * -p, --listen-port    The port clients should connect to.
069         * 
070         * This would be 20.
071         * </pre>
072         */
073        private final int indexOfDescriptionColumn;
074        private boolean needsNewline = false;
075        private StringBuilder builder = null;
076
077        Usage(Collection<Argument<?>> arguments, Locale locale, ProgramInformation program)
078        {
079                Collection<Argument<?>> visibleArguments = filter(arguments, IS_VISIBLE);
080                this.locale = locale;
081                this.argumentsToPrint = copyOf(sortedArguments(visibleArguments));
082                this.indexOfDescriptionColumn = determineLongestNameColumn() + SPACES_BETWEEN_COLUMNS;
083                this.program = program;
084        }
085
086        Usage(Collection<Argument<?>> arguments, Locale locale)
087        {
088                this(arguments, locale, ProgramInformation.AUTO);
089        }
090
091        String forCommand()
092        {
093                return commandUsage() + buildArgumentDescriptions();
094        }
095
096        private Iterable<Argument<?>> sortedArguments(Collection<Argument<?>> arguments)
097        {
098                Collection<Argument<?>> indexedArguments = filter(arguments, IS_INDEXED);
099                Iterable<Argument<?>> indexedWithoutVariableArity = filter(indexedArguments, not(IS_OF_VARIABLE_ARITY));
100                Iterable<Argument<?>> indexedWithVariableArity = filter(indexedArguments, IS_OF_VARIABLE_ARITY);
101
102                List<Argument<?>> sortedArgumentsByName = newArrayList(filter(arguments, not(IS_INDEXED)));
103
104                Collections.sort(sortedArgumentsByName, usingToString());
105
106                return Iterables.concat(indexedWithoutVariableArity, sortedArgumentsByName, indexedWithVariableArity);
107        }
108
109        private int determineLongestNameColumn()
110        {
111                int longestNameSoFar = 0;
112                for(Argument<?> arg : argumentsToPrint)
113                {
114                        longestNameSoFar = max(longestNameSoFar, lengthOfNameColumn(arg));
115                }
116                return longestNameSoFar;
117        }
118
119        private static int lengthOfNameColumn(final Argument<?> argument)
120        {
121                int namesLength = 0;
122
123                for(String name : argument.names())
124                {
125                        namesLength += name.length();
126                }
127                int separatorLength = max(0, UsageTexts.NAME_SEPARATOR.length() * (argument.names().size() - 1));
128
129                int metaLength = argument.metaDescriptionInLeftColumn().length();
130
131                return namesLength + separatorLength + metaLength;
132        }
133
134        private String buildArgumentDescriptions()
135        {
136                builder = newStringBuilder();
137                for(Argument<?> arg : argumentsToPrint)
138                {
139                        usageForArgument(arg);
140                }
141                return builder.toString();
142        }
143
144        private StringBuilder newStringBuilder()
145        {
146                // Two lines for each argument
147                return StringBuilders.withExpectedSize(2 * argumentsToPrint.size() * (indexOfDescriptionColumn + CHARACTERS_IN_AVERAGE_ARGUMENT_DESCRIPTION));
148        }
149
150        /**
151         * Returns the usage text that's suitable to print on {@link System#out}.
152         */
153        @Override
154        public String toString()
155        {
156                return mainUsage() + buildArgumentDescriptions();
157        }
158
159        private String mainUsage()
160        {
161                String mainUsage = UsageTexts.USAGE_HEADER + program.programName();
162
163                if(hasArguments())
164                {
165                        mainUsage += UsageTexts.ARGUMENT_INDICATOR;
166                }
167
168                mainUsage += NEWLINE + program.programDescription();
169
170                if(hasArguments())
171                {
172                        mainUsage += NEWLINE + UsageTexts.ARGUMENT_HEADER + NEWLINE;
173                }
174
175                return mainUsage;
176        }
177
178        private String commandUsage()
179        {
180                return hasArguments() ? UsageTexts.ARGUMENT_HEADER + NEWLINE : "";
181        }
182
183        private boolean hasArguments()
184        {
185                return !argumentsToPrint.isEmpty();
186        }
187
188        /**
189         * <pre>
190         *      -foo   Foo something [Required]
191         *              Valid values: 1 to 5
192         *        -bar   Bar something
193         *              Default: 0
194         * </pre>
195         */
196        private void usageForArgument(final Argument<?> arg)
197        {
198                int lengthBeforeCurrentArgument = builder.length();
199
200                NAME_JOINER.appendTo(builder, arg.names());
201
202                builder.append(arg.metaDescriptionInLeftColumn());
203
204                int lengthOfFirstColumn = builder.length() - lengthBeforeCurrentArgument;
205                builder.append(spaces(indexOfDescriptionColumn - lengthOfFirstColumn));
206
207                // TODO: handle long descriptions, names, meta descriptions, default value
208                // descriptions
209                String description = arg.description();
210                if(!description.isEmpty())
211                {
212                        builder.append(description);
213                        addIndicators(arg);
214                        needsNewline = true;
215                        newlineWithIndentation();
216                        valueExplanation(arg);
217                }
218                else
219                {
220                        valueExplanation(arg);
221                        addIndicators(arg);
222                }
223                builder.append(NEWLINE);
224                needsNewline = false;
225        }
226
227        private void newlineWithIndentation()
228        {
229                if(needsNewline)
230                {
231                        builder.append(NEWLINE);
232                        builder.append(spaces(indexOfDescriptionColumn));
233                        needsNewline = false;
234                }
235        }
236
237        private <T> void addIndicators(final Argument<T> arg)
238        {
239                if(arg.isRequired())
240                {
241                        builder.append(UsageTexts.REQUIRED);
242                }
243                if(arg.isAllowedToRepeat())
244                {
245                        builder.append(UsageTexts.ALLOWS_REPETITIONS);
246                }
247        }
248
249        private <T> void valueExplanation(final Argument<T> arg)
250        {
251                // TODO: handle long value explanations, replace each newline with enough spaces,
252                // split up long lines, Use BreakIterator.getLineInstance()?
253                String description = arg.descriptionOfValidValues(locale);
254                if(!description.isEmpty())
255                {
256                        boolean isCommand = arg.parser() instanceof Command;
257                        if(isCommand)
258                        {
259                                // For commands the validValues is a usage text itself for the command arguments
260                                // +1 = indentation so that command options are tucked under the command
261                                String spaces = spaces(indexOfDescriptionColumn + 1);
262                                description = description.replace(NEWLINE, NEWLINE + spaces);
263                        }
264                        else
265                        {
266                                String meta = arg.metaDescriptionInRightColumn();
267                                builder.append(meta + ": ");
268                        }
269
270                        builder.append(description);
271                        needsNewline = true;
272                }
273                if(arg.isRequired())
274                        return;
275
276                String descriptionOfDefaultValue = arg.defaultValueDescription(locale);
277                if(descriptionOfDefaultValue != null)
278                {
279                        newlineWithIndentation();
280                        String spaces = spaces(indexOfDescriptionColumn + UsageTexts.DEFAULT_VALUE_START.length());
281                        descriptionOfDefaultValue = descriptionOfDefaultValue.replace(NEWLINE, NEWLINE + spaces);
282
283                        builder.append(UsageTexts.DEFAULT_VALUE_START).append(descriptionOfDefaultValue);
284                }
285        }
