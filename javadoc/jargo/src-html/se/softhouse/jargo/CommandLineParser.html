017    import static com.google.common.base.Preconditions.checkNotNull;
018    import static java.util.Arrays.asList;
019    import static se.softhouse.jargo.Arguments.command;
020    
021    import java.util.Arrays;
022    import java.util.Collection;
023    import java.util.List;
024    import java.util.Locale;
025    import java.util.concurrent.atomic.AtomicReference;
026    import java.util.concurrent.locks.Lock;
027    import java.util.concurrent.locks.ReentrantLock;
028    
029    import javax.annotation.CheckReturnValue;
030    import javax.annotation.Nonnull;
031    import javax.annotation.concurrent.GuardedBy;
032    import javax.annotation.concurrent.ThreadSafe;
033    
034    import com.google.common.collect.Lists;
035    import com.google.common.util.concurrent.Atomics;

037    /**
038     * Manages multiple {@link Argument}s and/or {@link Command}s. The brain of this API.
039     * Its primary goal is to decide which {@link Argument} each input string belongs to.
040     * Immutability is dearly embraced. Thus a {@link CommandLineParser} can be reused to parse
041     * arguments over and over again. Different {@link CommandLineParser}s can even
042     * share {@link Argument} configurations as {@link Argument} instances are immutable as well.
043     * Documentation through example:
044     * 
045     * <pre class="prettyprint">
046     * <code class="language-java">
047     * import static se.softhouse.jargo.Arguments.*;
048     * ...
049     * String[] args = {"--enable-logging", "--listen-port", "8090", "Hello"};
050     * 
051     * Argument&lt;?&gt; helpArgument = helpArgument("-h", "--help"); //Will throw when -h is encountered
052     * Argument&lt;Boolean&gt; enableLogging = optionArgument("-l", "--enable-logging").description("Output debug information to standard out").build();
053     * Argument&lt;String&gt; greetingPhrase = stringArgument().description("A greeting phrase to greet new connections with").build();
054     * Argument&lt;List&lt;Integer&gt;&gt; ports = integerArgument("-p", "--listen-port")
055     *                                              .defaultValue(8080)
056     *                                              .description("The port clients should connect to.")
057     *                                              .metaDescription("&lt;port&gt;")
058     *                                              .limitTo(Range.closed(0, 65536))
059     *                                              .repeated().build();
060     * 
061     * try
062     * {
063     *   ParsedArguments arguments = CommandLineParser.withArguments(helpArgument, greetingPhrase, enableLogging, ports).parse(args);
064     *   assertThat(arguments.get(enableLogging)).isTrue();
065     *   assertThat(arguments.get(ports)).isEqualTo(Arrays.asList(8090));
066     *   assertThat(arguments.get(greetingPhrase)).isEqualTo("Hello");
067     * }
068     * catch(ArgumentException exception)
069     * {
070     *   System.out.println(exception.getMessageAndUsage());
071     *   System.exit(1);
072     * }
073     * </code>
074     * </pre>
075     * 
076     * <pre>
077     * For this program the usage would look like ("YourProgramName" is fetched from stack traces by default):
078     * <code>
079     * Usage: YourProgramName [Arguments]
080     * 
081     * Arguments:
082     * &lt;string&gt;                       A greeting phrase to greet new connections with
083     *                                &lt;string&gt;: any string
084     *                                Default:
085     * -l, --enable-logging           Output debug information to standard out
086     *                                Default: disabled
087     * -p, --listen-port &lt;port&gt;       The port clients should connect to [Supports Multiple occurrences]
088     *                                port: [0‥65536]
089     *                                Default: 8080
090     * </code>
091     * If something goes wrong during the parsing (Missing required arguments, Unexpected arguments, Invalid values),
092     * it will be described by the ArgumentException. Use {@link ArgumentException#getMessageAndUsage()} if you
093     * want to explain what went wrong to the user.
094     * </pre>
095     * 
096     * <b>Internationalization</b> By default {@link Locale#US} is used for parsing strings and printing
097     * usages. To change this use {@link #locale(Locale)}.<br>
098     * <b>Thread safety concerns:</b> If there is a parsing occurring while any modifying method is
099     * invoked, such as {@link #programName(String)}, the changes won't be seen until the start of the
100     * next parse invocation. Also, if the goal is to set {@link #programName(String)} and
101     * {@link #programDescription(String)} atomically you'll have to synchronize such changes
102     * externally.
103     */
104    @ThreadSafe
105    public final class CommandLineParser
106    {
107            // Internally CommandLineParser is a builder for CommandLineParserInstance but the idea is to
108            // keep this idea hidden in the API to lessen the API complexity
109    
110            static final Locale US_BY_DEFAULT = Locale.US;
111    
112            // TODO(jontejj): use getAndSet instead of set on Atomics
113    
114            @GuardedBy("modifyGuard") private final List<Argument<?>> argumentDefinitions;
115            @GuardedBy("modifyGuard") private final AtomicReference<ProgramInformation> programInformation = Atomics.newReference(ProgramInformation.AUTO);
116    
117            private final AtomicReference<Locale> locale = Atomics.newReference(US_BY_DEFAULT);
118    
119            @GuardedBy("modifyGuard") private final AtomicReference<CommandLineParserInstance> cachedParser;
120    
121            private final Lock modifyGuard = new ReentrantLock();
122    
123            /**
124             * Creates a {@link CommandLineParser} with support for the given {@code argumentDefinitions}.
125             * {@link Command}s can be added later with {@link #andCommands(Command...)}.
126             * 
127             * @param argumentDefinitions {@link Argument}s produced with {@link Arguments} or
128             *            with your own disciples of {@link ArgumentBuilder}
129             * @return a CommandLineParser which you can call {@link CommandLineParser#parse(String...)} on
130             *         and get {@link ParsedArguments} out of.
131             * @throws IllegalArgumentException if the given {@code argumentDefinitions} would create an
132             *             erroneous CommandLineParser
133             * @see CommandLineParser
134             */
135            @CheckReturnValue
136            @Nonnull
137            public static CommandLineParser withArguments(final Argument<?> ... argumentDefinitions)
138            {
139                    return new CommandLineParser(Arrays.asList(argumentDefinitions));
140            }
141    
142            /**
143             * {@link Iterable} version of {@link #withArguments(Argument...)}
144             */
145            @CheckReturnValue
146            @Nonnull
147            public static CommandLineParser withArguments(final Iterable<Argument<?>> argumentDefinitions)
148            {
149                    return new CommandLineParser(argumentDefinitions);
150            }
151    
152            /**
153             * Creates a {@link CommandLineParser} with support for {@code commands}. To add additional
154             * {@link Argument}s or {@link Command}s there is {@link #andArguments(Argument...)} and
155             * {@link #andCommands(Command...)}.
156             * 
157             * @param commands the commands to support initially
158             * @return a CommandLineParser which you can call {@link CommandLineParser#parse(String...)} on
159             *         and get {@link ParsedArguments} out of.
160             * @see CommandLineParser
161             */
162            @CheckReturnValue
163            @Nonnull
164            public static CommandLineParser withCommands(final Command ... commands)
165            {
166                    return new CommandLineParser(commandsToArguments(commands));
167            }
168    
169            /**
170             * Parses {@code actualArguments} (typically from the command line, i.e argv) and returns the
171             * parsed values in a {@link ParsedArguments} container. {@link Locale#US} is used to parse
172             * strings by default. Use {@link #locale(Locale)} to change it.
173             * 
174             * @throws ArgumentException if an invalid argument is encountered during the parsing
175             */
176            @Nonnull
177            public ParsedArguments parse(final String ... actualArguments) throws ArgumentException
178            {
179                    return parser().parse(asList(actualArguments), locale());
180            }
181    
182            /**
183             * {@link Iterable} version of {@link #parse(String...)}
184             */
185            @Nonnull
186            public ParsedArguments parse(final Iterable<String> actualArguments) throws ArgumentException
187            {
188                    return parser().parse(actualArguments, locale());
189            }
190    
191            /**
192             * Returns a usage text describing all arguments this {@link CommandLineParser} handles.
193             * Suitable to print on {@link System#out}.
194             */
195            @CheckReturnValue
196            @Nonnull
197            public Usage usage()
198            {
199                    return new Usage(parser().allArguments(), locale(), programInformation(), false);
200            }
201    
202            /**
203             * Adds support for {@code commandsToAlsoSupport} in this {@link CommandLineParser}. Typically
204             * used in a chained fashion when faced with many supported {@link Command}s.
205             * 
206             * @param commandsToAlsoSupport the commands to add support for
207             * @return this {@link CommandLineParser} to allow for chained calls
208             */
209            @CheckReturnValue
210            @Nonnull
211            public CommandLineParser andCommands(final Command ... commandsToAlsoSupport)
212            {
213                    verifiedAdd(commandsToArguments(commandsToAlsoSupport));
214                    return this;
215            }
216    
217            /**
218             * <pre>
219             * Adds support for {@code argumentsToAlsoSupport} in this {@link CommandLineParser}.
220             * Typically used in a chained fashion when faced with many supported arguments.
221             * Another usage is simply for readability: group arguments by logical groups
222             * 
223             * @param argumentsToAlsoSupport the arguments to add support for
224             * @return this {@link CommandLineParser} to allow for chained calls
225             * </pre>
226             */
227            @CheckReturnValue
228            @Nonnull
229            public CommandLineParser andArguments(final Argument<?> ... argumentsToAlsoSupport)
230            {
231                    verifiedAdd(asList(argumentsToAlsoSupport));
232                    return this;
233            }
234    
235            /**
236             * Verify that the parser would be usable after adding {@code argumentsToAdd} to it.
237             * This allows future parse operations to still use the unaffected old parser.
238             */
239            private void verifiedAdd(Collection<Argument<?>> argumentsToAdd)
240            {
241                    try
242                    {
243                            modifyGuard.lock();
244                            List<Argument<?>> newDefinitions = Lists.newArrayList(argumentDefinitions);
245                            newDefinitions.addAll(argumentsToAdd);
246                            cachedParser.set(new CommandLineParserInstance(newDefinitions, programInformation()));
247                            // Everything went fine, accept argument
248                            argumentDefinitions.addAll(argumentsToAdd);
249                    }
250                    finally
251                    {
252                            modifyGuard.unlock();
253                    }
254            }
255    
256            /**
257             * Sets the {@code programName} to print with {@link #usage()}
258             * 
259             * @return this parser
260             */
261            public CommandLineParser programName(String programName)
262            {
263                    try
264                    {
265                            modifyGuard.lock();
266                            programInformation.set(programInformation().programName(programName));
267                            cachedParser.set(new CommandLineParserInstance(argumentDefinitions, programInformation()));
268                    }
269                    finally
270                    {
271                            modifyGuard.unlock();
272                    }
273                    return this;
274            }
275    
276            /**
277             * Sets the {@code programDescription} to print with {@link #usage()}
278             * 
279             * @return this parser
280             */
281            public CommandLineParser programDescription(String programDescription)
282            {
283                    try
284                    {
285                            modifyGuard.lock();
286                            programInformation.set(programInformation().programDescription(programDescription));
287                            cachedParser.set(new CommandLineParserInstance(argumentDefinitions, programInformation()));
288                    }
289                    finally
290                    {
291                            modifyGuard.unlock();
292                    }
293                    return this;
294            }
295    
296            /**
297             * {@link Locale#getDefault()} is dangerous to use. It's a global and mutable variable
298             * which makes it hard to give any guarantees on what it will be at a certain time. This is why
299             * {@link Locale#US} is used by default instead.
300             * If {@link Locale#getDefault()} is wanted, use {@link #locale(Locale)
301             * locale(Locale.getDefault())}. To use a different {@link Locale} for <b>only</b> one of your
302             * {@link Argument}s you can override the value set here with
303             * {@link ArgumentBuilder#locale(Locale)}.
304             * 
305             * @param localeToUse the {@link Locale} to parse input strings with (it will be passed to
306             *            {@link StringParser#parse(String, Locale)} and
307             *            {@link StringParser#descriptionOfValidValues(Locale)})
308             * @return this parser
309             */
310            public CommandLineParser locale(Locale localeToUse)
311            {
312                    locale.set(checkNotNull(localeToUse));
313                    return this;
314            }
315    
316            /**
317             * Returns the {@link #usage()} for this {@link CommandLineParser}
318             */
319            @Override
320            public String toString()
321            {
322                    return usage().toString();
323            }
324    
325            CommandLineParser(Iterable<Argument<?>> argumentDefinitions)
326            {
327                    this.argumentDefinitions = Lists.newArrayList(argumentDefinitions);
328                    this.cachedParser = Atomics.newReference(new CommandLineParserInstance(this.argumentDefinitions, programInformation()));
329            }
330    
331            CommandLineParserInstance parser()
332            {
333                    return cachedParser.get();
334            }
335    
336            Locale locale()
337            {
338                    return locale.get();
339            }
340    
341            ProgramInformation programInformation()
342            {
343                    return programInformation.get();
344            }
345    
346            private static List<Argument<?>> commandsToArguments(final Command ... commands)
347            {
348                    List<Argument<?>> commandsAsArguments = Lists.newArrayListWithExpectedSize(commands.length);
349                    for(Command c : commands)
350                    {
351                            commandsAsArguments.add(command(c).build());
352                    }
353                    return commandsAsArguments;
354            }
355    }
