018import static java.util.Arrays.asList;
019import static se.softhouse.jargo.Arguments.command;

021import java.util.Arrays;
022import java.util.Collection;
023import java.util.List;
024import java.util.Locale;

037/**
038 * Manages multiple {@link Argument}s and/or {@link Command}s. The brain of this API.
039 * Its primary goal is to decide which {@link Argument} each input string belongs to.
040 * Immutability is dearly embraced. Thus a {@link CommandLineParser} can be reused to parse
041 * arguments over and over again. Different {@link CommandLineParser}s can even
042 * share {@link Argument} configurations as {@link Argument} instances are immutable as well.
043 * Documentation through example:
044 * 
045 * <pre class="prettyprint">
046 * <code class="language-java">
047 * import static se.softhouse.jargo.Arguments.*;
048 * ...
049 * String[] args = {"--enable-logging", "--listen-port", "8090", "Hello"};
050 * 
051 * Argument&lt;Boolean&gt; enableLogging = optionArgument("-l", "--enable-logging").description("Output debug information to standard out").build();
052 * Argument&lt;String&gt; greetingPhrase = stringArgument().description("A greeting phrase to greet new connections with").build();
053 * Argument&lt;List&lt;Integer&gt;&gt; ports = integerArgument("-p", "--listen-port")
054 *                                              .defaultValue(8080)
055 *                                              .description("The port clients should connect to.")
056 *                                              .metaDescription("&lt;port&gt;")
057 *                                              .limitTo(Ranges.closed(0, 65536)
058 *                                              .repeated().build();
059 * 
060 * try
061 * {
062 *   ParsedArguments arguments = CommandLineParser.withArguments(greetingPhrase, enableLogging, ports).parse(args);
063 *   assertThat(arguments.get(enableLogging)).isTrue();
064 *   assertThat(arguments.get(ports)).isEqualTo(Arrays.asList(8090));
065 *   assertThat(arguments.get(greetingPhrase)).isEqualTo("Hello");
066 * }
067 * catch(ArgumentException exception)
068 * {
069 *   System.out.println(exception.getMessageAndUsage());
070 *   System.exit(1);
071 * }
072 * </code>
073 * </pre>
074 * 
075 * <pre>
076 * For this program the usage would look like ("YourProgramName" is fetched from stack traces by default):
077 * <code>
078 * Usage: YourProgramName [Arguments]
079 * 
080 * Arguments:
081 * &lt;string&gt;                       A greeting phrase to greet new connections with
082 *                                &lt;string&gt;: any string
083 *                                Default:
084 * -l, --enable-logging           Output debug information to standard out
085 *                                Default: disabled
086 * -p, --listen-port &lt;port&gt;       The port clients should connect to [Supports Multiple occurrences]
087 *                                port: [0‥65536]
088 *                                Default: 8080
089 * </code>
090 * If something goes wrong during the parsing (Missing required arguments, Unexpected arguments, Invalid values),
091 * it will be described by the ArgumentException. Use {@link ArgumentException#getMessageAndUsage()} if you
092 * want to explain what went wrong to the user.
093 * </pre>
094 * 
095 * <b>Internationalization</b> By default {@link Locale#US} is used for parsing strings and printing
096 * usages. To change this use {@link #locale(Locale)}.<br>
097 * <b>Thread safety concerns:</b> If there is a parsing occurring while any modifying method is
098 * invoked, such as {@link #programName(String)}, the changes won't be seen until the start of the
099 * next parse invocation. Also, if the goal is to set {@link #programName(String)} and
100 * {@link #programDescription(String)} atomically you'll have to synchronize such changes
101 * externally.
102 */
103@ThreadSafe
104public final class CommandLineParser
105{
106        // Internally CommandLineParser is a builder for CommandLineParserInstance but the idea is to
107        // keep this idea hidden in the API to lessen the API complexity
108
109        static final Locale US_BY_DEFAULT = Locale.US;
110
111        @GuardedBy("modifyGuard") private final List<Argument<?>> argumentDefinitions;
112        @GuardedBy("modifyGuard") private final AtomicReference<ProgramInformation> programInformation = Atomics.newReference(ProgramInformation.AUTO);
113
114        private final AtomicReference<Locale> locale = Atomics.newReference(US_BY_DEFAULT);
115
116        @GuardedBy("modifyGuard") private final AtomicReference<CommandLineParserInstance> cachedParser;
117
118        private final Lock modifyGuard = new ReentrantLock();
119
120        /**
121         * Creates a {@link CommandLineParser} with support for the given {@code argumentDefinitions}.
122         * {@link Command}s can be added later with {@link #andCommands(Command...)}.
123         * 
124         * @param argumentDefinitions {@link Argument}s produced with {@link Arguments} or
125         *            with your own disciples of {@link ArgumentBuilder}
126         * @return a CommandLineParser which you can call {@link CommandLineParser#parse(String...)} on
127         *         and get {@link ParsedArguments} out of.
128         * @throws IllegalArgumentException if the given {@code argumentDefinitions} would create an
129         *             erroneous CommandLineParser
130         * @see CommandLineParser
131         */
132        @CheckReturnValue
133        @Nonnull
134        public static CommandLineParser withArguments(final Argument<?> ... argumentDefinitions)
135        {
136                return new CommandLineParser(Arrays.asList(argumentDefinitions));
137        }
138
139        /**
140         * {@link Iterable} version of {@link #withArguments(Argument...)}
141         */
142        @CheckReturnValue
143        @Nonnull
144        public static CommandLineParser withArguments(final Iterable<Argument<?>> argumentDefinitions)
145        {
146                return new CommandLineParser(argumentDefinitions);
147        }
148
149        /**
150         * Creates a {@link CommandLineParser} with support for {@code commands}. To add additional
151         * {@link Argument}s or {@link Command}s there is {@link #andArguments(Argument...)} and
152         * {@link #andCommands(Command...)}.
153         * 
154         * @param commands the commands to support initially
155         * @return a CommandLineParser which you can call {@link CommandLineParser#parse(String...)} on
156         *         and get {@link ParsedArguments} out of.
157         * @see CommandLineParser
158         */
159        @CheckReturnValue
160        @Nonnull
161        public static CommandLineParser withCommands(final Command ... commands)
162        {
163                return new CommandLineParser(commandsToArguments(commands));
164        }
165
166        /**
167         * Parses {@code actualArguments} (typically from the command line, i.e argv) and returns the
168         * parsed values in a {@link ParsedArguments} container. {@link Locale#US} is used to parse
169         * strings by default. Use {@link #locale(Locale)} to change it.
170         * 
171         * @throws ArgumentException if an invalid argument is encountered during the parsing
172         */
173        @Nonnull
174        public ParsedArguments parse(final String ... actualArguments) throws ArgumentException
175        {
176                return parser().parse(Lists.newArrayList(actualArguments), locale());
177        }
178
179        /**
180         * {@link Iterable} version of {@link #parse(String...)}
181         */
182        @Nonnull
183        public ParsedArguments parse(final Iterable<String> actualArguments) throws ArgumentException
184        {
185                return parser().parse(Lists.newArrayList(actualArguments), locale());
186        }
187
188        /**
189         * Returns a usage text describing all arguments this {@link CommandLineParser} handles.
190         * Suitable to print on {@link System#out}.
191         */
192        @CheckReturnValue
193        @Nonnull
194        public Usage usage()
195        {
196                return new Usage(parser().allArguments(), locale(), programInformation());
197        }
198
199        /**
200         * Adds support for {@code commandsToAlsoSupport} in this {@link CommandLineParser}. Typically
201         * used in a chained fashion when faced with many supported {@link Command}s.
202         * 
203         * @param commandsToAlsoSupport the commands to add support for
204         * @return this {@link CommandLineParser} to allow for chained calls
205         */
206        @CheckReturnValue
207        @Nonnull
208        public CommandLineParser andCommands(final Command ... commandsToAlsoSupport)
209        {
210                verifiedAdd(commandsToArguments(commandsToAlsoSupport));
211                return this;
212        }
213
214        /**
215         * Adds support for {@code argumentsToAlsoSupport} in this {@link CommandLineParser}. Typically
216         * used in a chained fashion when faced with many supported arguments.
217         * 
218         * @param argumentsToAlsoSupport the arguments to add support for
219         * @return this {@link CommandLineParser} to allow for chained calls
220         */
221        @CheckReturnValue
222        @Nonnull
223        public CommandLineParser andArguments(final Argument<?> ... argumentsToAlsoSupport)
224        {
225                // TODO: test thread safety, sleep long in a parser and, add argument and verify that the
226                // second parse uses the added argument
227                verifiedAdd(asList(argumentsToAlsoSupport));
228                return this;
229        }
230
231        /**
232         * Verify that the parser would be usable after adding {@code argumentsToAdd} to it.
233         * This allows future parse operations to still use the unaffected old parser.
234         */
235        private void verifiedAdd(Collection<Argument<?>> argumentsToAdd)
236        {
237                try
238                {
239                        modifyGuard.lock();
240                        List<Argument<?>> newDefinitions = Lists.newArrayList(argumentDefinitions);
241                        newDefinitions.addAll(argumentsToAdd);
242                        cachedParser.set(new CommandLineParserInstance(newDefinitions, programInformation()));
243                        // Everything went fine, accept argument
244                        argumentDefinitions.addAll(argumentsToAdd);
245                }
246                finally
247                {
248                        modifyGuard.unlock();
249                }
250        }
251
252        /**
253         * Sets the {@code programName} to print with {@link #usage()}
254         * 
255         * @return this parser
256         */
257        public CommandLineParser programName(String programName)
258        {
259                try
260                {
261                        modifyGuard.lock();
262                        programInformation.set(programInformation().programName(programName));
263                        cachedParser.set(new CommandLineParserInstance(argumentDefinitions, programInformation()));
264                }
265                finally
266                {
267                        modifyGuard.unlock();
268                }
269                return this;
270        }
271
272        /**
273         * Sets the {@code programDescription} to print with {@link #usage()}
274         * 
275         * @return this parser
276         */
277        public CommandLineParser programDescription(String programDescription)
278        {
279                try
280                {
281                        modifyGuard.lock();
282                        programInformation.set(programInformation().programDescription(programDescription));
283                        cachedParser.set(new CommandLineParserInstance(argumentDefinitions, programInformation()));
284                }
285                finally
286                {
287                        modifyGuard.unlock();
288                }
289                return this;
290        }
291
292        /**
293         * {@link Locale#getDefault()} is dangerous to use. It's a global and mutable variable
294         * which makes it hard to give any guarantees on what it will be at a certain time. This is why
295         * {@link Locale#US} is used by default instead.
296         * If {@link Locale#getDefault()} is wanted, use {@link #locale(Locale)
297         * locale(Locale.getDefault())}. To use a different {@link Locale} for <b>only</b> one of your
298         * {@link Argument}s you can override the value set here with
299         * {@link ArgumentBuilder#locale(Locale)}.
300         * 
301         * @param localeToUse the {@link Locale} to parse input strings with (it will be passed to
302         *            {@link StringParser#parse(String, Locale)} and
303         *            {@link StringParser#descriptionOfValidValues(Locale)})
304         * @return this parser
305         */
306        public CommandLineParser locale(Locale localeToUse)
307        {
308                locale.set(checkNotNull(localeToUse));
309                return this;
310        }
311
312        /**
313         * Returns the {@link #usage()} for this {@link CommandLineParser}
314         */
315        @Override
316        public String toString()
317        {
318                return usage().toString();
319        }
320
321        CommandLineParser(Iterable<Argument<?>> argumentDefinitions)
322        {
323                this.argumentDefinitions = Lists.newArrayList(argumentDefinitions);
324                this.cachedParser = Atomics.newReference(new CommandLineParserInstance(this.argumentDefinitions, programInformation()));
325        }
326
327        CommandLineParserInstance parser()
328        {
329                return cachedParser.get();
330        }
331
332        Locale locale()
333        {
334                return locale.get();
335        }
336
337        ProgramInformation programInformation()
338        {
339                return programInformation.get();
340        }
341
342        private static List<Argument<?>> commandsToArguments(final Command ... commands)
343        {
344                List<Argument<?>> commandsAsArguments = Lists.newArrayListWithExpectedSize(commands.length);
345                for(Command c : commands)
346                {
347                        commandsAsArguments.add(command(c).build());
348                }
349                return commandsAsArguments;
350        }
