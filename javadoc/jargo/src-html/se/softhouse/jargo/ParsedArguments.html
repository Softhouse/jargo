015    package se.softhouse.jargo;
016    
017    import static com.google.common.base.Preconditions.checkArgument;
018    import static com.google.common.base.Preconditions.checkNotNull;
019    import static com.google.common.base.Predicates.and;
020    import static com.google.common.base.Predicates.in;
021    import static com.google.common.base.Predicates.not;
022    import static com.google.common.collect.Collections2.filter;
023    import static com.google.common.collect.Maps.newLinkedHashMap;
024    import static se.softhouse.jargo.Argument.IS_REQUIRED;
025    
026    import java.util.Collection;
027    import java.util.Locale;
028    import java.util.Map;
029    import java.util.Set;
030    
031    import javax.annotation.CheckReturnValue;
032    import javax.annotation.Nonnull;
033    import javax.annotation.Nullable;
034    import javax.annotation.concurrent.Immutable;
035    
036    import se.softhouse.jargo.internal.Texts.ProgrammaticErrors;
037    
038    import com.google.common.collect.Sets;
039    
040    /**
041     * Holds parsed arguments for a {@link CommandLineParser#parse(String...)} invocation.
042     * Use {@link #get(Argument)} to fetch a parsed command line value.
043     */
044    @Immutable
045    public final class ParsedArguments
046    {
047            /**
048             * Stores results from {@link StringParser#parse(String, Locale)}
049             */
050            @Nonnull private final Map<Argument<?>, Object> parsedArguments = newLinkedHashMap();
051            @Nonnull private final Set<Argument<?>> allArguments;
052            /**
053             * Keeps a running total of how many indexed arguments that have been parsed
054             */
055            private int indexedArgumentsParsed = 0;
056    
057            ParsedArguments(Set<Argument<?>> arguments)
058            {
059                    allArguments = arguments;
060            }
061    
062            /**
063             * Returns the parsed value for the given {@code argumentToFetch},
064             * if no value was given on the command line the default value is
065             * returned.
066             * 
067             * @see StringParser#defaultValue()
068             */
069            @Nullable
070            @CheckReturnValue
071            public <T> T get(final Argument<T> argumentToFetch)
072            {
073                    if(!wasGiven(argumentToFetch))
074                    {
075                            checkArgument(allArguments.contains(argumentToFetch), ProgrammaticErrors.ILLEGAL_ARGUMENT, argumentToFetch);
076                            return argumentToFetch.defaultValue();
077                    }
078                    return getValue(argumentToFetch);
079            }
080    
081            /**
082             * Checks if an {@link Argument} was given in the command line arguments
083             * 
084             * @param argument the {@link Argument} to check
085             * @return true if {@code argument} was given, that is, the default value will not be used
086             */
087            public boolean wasGiven(Argument<?> argument)
088            {
089                    return parsedArguments.containsKey(checkNotNull(argument));
090            }
091    
092            @Override
093            public String toString()
094            {
095                    return parsedArguments.toString();
096            }
097    
098            @Override
099            public boolean equals(@Nullable Object other)
100            {
101                    if(this == other)
102                            return true;
103                    if(!(other instanceof ParsedArguments))
104                            return false;
105    
106                    return parsedArguments.equals(((ParsedArguments) other).parsedArguments);
107            }
108    
109            @Override
110            public int hashCode()
111            {
112                    return parsedArguments.hashCode();
113            }
114    
115            // Publicly this class is Immutable, CommandLineParserInstance is only allowed to modify it
116            // during parsing
117    
118            <T> void put(final Argument<T> definition, @Nullable final T value)
119            {
120                    if(definition.isIndexed())
121                    {
122                            indexedArgumentsParsed++;
123                    }
124                    parsedArguments.put(definition, value);
125            }
126    
127            <T> T getValue(final Argument<T> definition)
128            {
129                    // Safe because put guarantees that the map is heterogeneous
130                    @SuppressWarnings("unchecked")
131                    T value = (T) parsedArguments.get(definition);
132                    return value;
133            }
134    
135            Collection<Argument<?>> requiredArgumentsLeft()
136            {
137                    return filter(allArguments, and(not(in(parsedArguments.keySet())), IS_REQUIRED));
138            }
139    
140            int indexedArgumentsParsed()
141            {
142                    return indexedArgumentsParsed;
143            }
144    
145            Set<Argument<?>> parsedArguments()
146            {
147                    return parsedArguments.keySet();
148            }
149    
150            Set<String> nonParsedArguments()
151            {
152                    Set<String> validArguments = Sets.newHashSetWithExpectedSize(allArguments.size());
153                    for(Argument<?> argument : allArguments)
154                    {
155                            boolean wasGiven = wasGiven(argument);
156                            if(!wasGiven || argument.isAllowedToRepeat())
157                            {
158                                    for(String name : argument.names())
159                                    {
160                                            if(argument.separator().equals(ArgumentBuilder.DEFAULT_SEPARATOR) || argument.isPropertyMap())
161                                            {
162                                                    validArguments.add(name);
163                                            }
164                                            else
165                                            {
166                                                    validArguments.add(name + argument.separator());
167                                            }
168    
169                                    }
170                            }
171                    }
172                    return validArguments;
173            }
174    }
