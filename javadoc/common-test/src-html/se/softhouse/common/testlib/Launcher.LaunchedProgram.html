015    package se.softhouse.common.testlib;
016    
017    import static com.google.common.base.Preconditions.checkArgument;
018    import static com.google.common.base.Preconditions.checkNotNull;
019    import static com.google.common.io.ByteStreams.toByteArray;
020    import static java.lang.reflect.Modifier.isPublic;
021    import static java.lang.reflect.Modifier.isStatic;
022    
023    import java.io.File;
024    import java.io.IOException;
025    import java.util.Arrays;
026    import java.util.List;
027    
028    import com.google.common.base.Charsets;
029    import com.google.common.collect.Lists;
030    
031    /**
032     * Can launch java programs for a {@link Class} with a main method. Output is captured in the
033     * returned {@link LaunchedProgram}.
034     */
035    public final class Launcher
036    {
037            /**
038             * Result from a {@link Launcher#launch(Class, String...) launched program}
039             */
040            public static final class LaunchedProgram
041            {
042                    private final String errors;
043                    private final String output;
044                    private final String debugInformation;
045    
046                    private LaunchedProgram(String errors, String output, String debugInformation)
047                    {
048                            this.errors = errors;
049                            this.output = output;
050                            this.debugInformation = debugInformation;
051                    }
052    
053                    /**
054                     * The {@link System#out} in {@link Charsets#UTF_8 UTF-8} from the launched program
055                     */
056                    public String output()
057                    {
058                            return output;
059                    }
060    
061                    /**
062                     * The {@link System#err} in {@link Charsets#UTF_8 UTF-8} from the launched program
063                     */
064                    public String errors()
065                    {
066                            return errors;
067                    }
068    
069                    /**
070                     * Returns information suitable to print in case of errors on a debug level
071                     */
072                    public String debugInformation()
073                    {
074                            return debugInformation;
075                    }
076    
077                    @Override
078                    public String toString()
079                    {
080                            return "Output: " + output() + "\nErrors: " + errors();
081                    }
082            }
083    
084            private Launcher()
085            {
086            }
087    
088            /**
089             * Runs {@code classWithMainMethod} in a separate process using the system property
090             * {@code java.home} to find java and {@code java.class.path} for the class path. This method
091             * will wait until program execution has finished.
092             * 
093             * @param classWithMainMethod a class with a static "main" method
094             * @param programArguments optional arguments to pass to the program
095             * @return output/errors from the executed program
096             * @throws IOException if an I/O error occurs while starting {@code classWithMainMethod} as a
097             *             process
098             * @throws InterruptedException if the thread starting the program is
099             *             {@link Thread#interrupted()} while waiting for the program to finish
100             * @throws IllegalArgumentException if {@code classWithMainMethod} doesn't have a public static
101             *             main method
102             * @throws SecurityException if it's not possible to validate the existence of a main method in
103             *             {@code classWithMainMethod} (or if {@link SecurityManager#checkExec checkExec}
104             *             fails)
105             */
106            public static LaunchedProgram launch(Class<?> classWithMainMethod, String ... programArguments) throws IOException, InterruptedException
107            {
108                    checkNotNull(programArguments);
109                    try
110                    {
111    
112                            int modifiers = classWithMainMethod.getDeclaredMethod("main", String[].class).getModifiers();
113                            boolean validModifiers = isStatic(modifiers) && isPublic(modifiers);
114                            checkArgument(validModifiers, "%s's main method needs to be static and public for it to be launchable", classWithMainMethod.getName());
115                    }
116                    catch(NoSuchMethodException e)
117                    {
118                            throw new IllegalArgumentException("No main method found on: " + classWithMainMethod.getName(), e);
119                    }
120                    String jvm = new File(new File(System.getProperty("java.home"), "bin"), "java").getAbsolutePath();
121                    String classPath = System.getProperty("java.class.path");
122    
123                    List<String> args = Lists.newArrayList(jvm, "-cp", classPath, classWithMainMethod.getName());
124                    args.addAll(Arrays.asList(programArguments));
125    
126                    String debugInformation = "\njvm: " + jvm + "\nclasspath: " + classPath;
127                    return execute(args, debugInformation);
128            }
129    
130            /**
131             * Runs {@code program} with {@code programArguments} as arguments. This method
132             * will wait until program execution has finished.
133             * 
134             * @param program the executable to run
135             * @param programArguments optional arguments to pass to the program
136             * @return output/errors from the executed program
137             * @throws IOException if an I/O error occurs while starting {@code program}
138             * @throws InterruptedException if the thread starting the program is
139             *             {@link Thread#interrupted()} while waiting for the program to finish
140             * @throws SecurityException if {@link SecurityManager#checkExec checkExec} fails
141             */
142            public static LaunchedProgram launch(String program, String ... programArguments) throws IOException, InterruptedException
143            {
144                    return execute(Lists.asList(program, programArguments), "Failed to launch " + program);
145            }
146    
147            private static LaunchedProgram execute(List<String> args, String debugInformation) throws IOException, InterruptedException
148            {
149                    Process program = new ProcessBuilder().command(args).start();
150                    // TODO(jontejj): read streams simultaneously to avoid deadlock in the launched program
151                    // http://www.javaworld.com/javaworld/jw-12-2000/jw-1229-traps.html?page=4
152                    String output = new String(toByteArray(program.getInputStream()), Charsets.UTF_8);
153                    String errors = new String(toByteArray(program.getErrorStream()), Charsets.UTF_8);
154    
155                    program.waitFor();
156                    return new LaunchedProgram(errors, output, debugInformation);
157            }
158    }
