015    package se.softhouse.common.testlib;
016    
017    import static com.google.common.base.Preconditions.checkNotNull;
018    import static java.lang.Thread.currentThread;
019    import static java.util.concurrent.TimeUnit.MILLISECONDS;
020    import static se.softhouse.common.testlib.Constants.ONE_MINUTE;
021    
022    import java.util.Locale;
023    import java.util.concurrent.TimeUnit;
024    import java.util.concurrent.locks.ReentrantLock;
025    
026    import javax.annotation.concurrent.ThreadSafe;
027    
028    /**
029     * <pre>
030     * Utilities for testing different {@link Locale}s.
031     * </pre>
032     */
033    @ThreadSafe
034    public final class Locales
035    {
036            private Locales()
037            {
038            }
039    
040            /**
041             * The {@link Locale} representing Sweden
042             */
043            public static final Locale SWEDISH = new Locale("sv", "SE");
044    
045            /**
046             * See <a href="http://java.sys-con.com/node/46241">Turkish Java Needs Special Brewing</a> and
047             * <a href="http://www.codinghorror.com/blog/2008/03/whats-wrong-with-turkey.html">What's Wrong
048             * With Turkey?</a> for more details about why the Turkish {@link Locale} is suitable for
049             * testing purposes.
050             */
051            public static final Locale TURKISH = new Locale("tr", "TR");
052    
053            private static final ReentrantLock LOCK = new ReentrantLock();
054    
055            private static volatile Locale defaultLocale = Locale.getDefault();
056    
057            /**
058             * How long the previous caller said he wanted to use {@link Locale#getDefault()} for in
059             * milliseconds
060             */
061            private static volatile long maximumUsageTime = ONE_MINUTE;
062    
063            /**
064             * <pre>
065             *  Makes it possible to test invalid usages of {@link Locale#getDefault()} in a synchronized
066             *  manner making sure that test cases don't fail suddenly when they are run concurrently.
067             *  I.e a pipelined version of {@link Locale#getDefault()}.
068             * 
069             * You're allowed to use {@link Locale#getDefault()} for 60 seconds then other threads
070             * will start to receive {@link InterruptedException} because they can't get access to it.
071             * If you need more than 60 seconds use {@link #setDefault(Locale, long, TimeUnit)} instead.
072             * <b>Note:</b> Don't forget to call {@link #resetDefaultLocale()} when you're done with using
073             * the {@link Locale#getDefault()} method.
074             * 
075             * @param locale the locale {@link Locale#getDefault()} should return
076             * @throws InterruptedException if it wasn't possible to invoke
077             *             {@link Locale#setDefault(Locale)} due to another thread using the default
078             *             locale for longer than he specified (default 60 seconds)
079             * </pre>
080             */
081            public static void setDefault(Locale locale) throws InterruptedException
082            {
083                    setDefault(locale, 60, TimeUnit.SECONDS);
084            }
085    
086            /**
087             * <pre>
088             * Works like {@link #setDefault(Locale)} but it allows you to use {@link Locale#getDefault()}
089             * for the time specified by {@code time} and {@code unit}.
090             * <b>Note:</b> if the current thread already "owns" the default locale (by having called
091             * {@link #setDefault(Locale, long, TimeUnit)} before) {@code time} and {@code unit} will be
092             * ignored.
093             * 
094             * @param locale the locale {@link Locale#getDefault()} should return
095             * @param time the maximum time to use {@link Locale#getDefault()}
096             * @param unit the {@link TimeUnit} of the time argument
097             * @throws InterruptedException if it wasn't possible to invoke
098             *             {@link Locale#setDefault(Locale)} due to another thread using the default
099             *             locale for longer than he specified (default 60 seconds)
100             * </pre>
101             */
102            public static void setDefault(Locale locale, long time, TimeUnit unit) throws InterruptedException
103            {
104                    checkNotNull(locale);
105                    checkNotNull(unit);
106    
107                    // Allow the same thread to change the locale during his "period" without locking
108                    if(!LOCK.isHeldByCurrentThread())
109                    {
110                            if(!LOCK.tryLock(maximumUsageTime, TimeUnit.MILLISECONDS))
111                                    throw new InterruptedException("Thread (" + currentThread().getName() + ") Waited for " + MILLISECONDS.toSeconds(maximumUsageTime)
112                                                    + " seconds on " + LOCK + " to finish using " + Locale.getDefault() + " but timed out");
113                            maximumUsageTime = unit.toMillis(time);
114                            defaultLocale = Locale.getDefault();
115                    }
116    
117                    try
118                    {
119                            Locale.setDefault(locale);
120                    }
121                    catch(SecurityException e)
122                    {
123                            LOCK.unlock(); // Let someone else try again
124                            throw e;
125                    }
126            }
127    
128            /**
129             * Resets {@link Locale#getDefault()} to what it was when {@link #setDefault(Locale)} was
130             * called. Should be called after you're done with using {@link Locale#getDefault()}, typically
131             * called in a "tear down" method.
132             * 
133             * @return the default {@link Locale} that was set, or null if it wasn't this thread that owned
134             *         the default locale
135             */
136            public static Locale resetDefaultLocale()
137            {
138                    if(LOCK.isHeldByCurrentThread())
139                    {
140                            try
141                            {
142                                    Locale.setDefault(defaultLocale);
143                                    return defaultLocale;
144                            }
145                            finally
146                            {
147                                    LOCK.unlock();
148                            }
149                    }
150                    return null;
151            }
152    }
