015    package se.softhouse.common.testlib;
016    
017    import static com.google.common.util.concurrent.MoreExecutors.listeningDecorator;
018    import static java.util.concurrent.Executors.newSingleThreadExecutor;
019    
020    import java.util.concurrent.Callable;
021    import java.util.concurrent.Executor;
022    import java.util.concurrent.Future;
023    
024    import com.google.common.util.concurrent.FutureCallback;
025    import com.google.common.util.concurrent.Futures;
026    import com.google.common.util.concurrent.ListenableFuture;
027    import com.google.common.util.concurrent.ListeningExecutorService;
028    
029    /**
030     * Utility for running things concurrently.
031     */
032    public final class Concurrently
033    {
034            private Concurrently()
035            {
036            }
037    
038            /**
039             * Returns a {@link ListenableFuture} running {@code callable} asynchronously. When this method
040             * returns the {@link Callable#call()} may have been run or it may not. Call
041             * {@link Future#get()} on the returned {@link Future} to get the result from the calculation.
042             * The {@link Executor} running the returned {@link Future} is automatically shutdown when it's
043             * done. If threads should be reused and performance is critical you should roll your own and
044             * reuse the executor instead.
045             */
046            public static <T> ListenableFuture<T> run(Callable<T> callable)
047            {
048                    final ListeningExecutorService executor = listeningDecorator(newSingleThreadExecutor());
049                    ListenableFuture<T> futureForCallable = executor.submit(callable);
050                    Futures.addCallback(futureForCallable, new FutureCallback<T>(){
051                            @Override
052                            public void onSuccess(T result)
053                            {
054                                    executor.shutdownNow();
055                            }
056    
057                            @Override
058                            public void onFailure(Throwable t)
059                            {
060                                    executor.shutdownNow();
061                            }
062                    });
063                    return futureForCallable;
064            }
065    }
