015package se.softhouse.common.testlib;
016
017import static org.fest.assertions.Assertions.assertThat;
018
019import java.lang.reflect.Constructor;
020import java.lang.reflect.InvocationTargetException;
021import java.lang.reflect.Modifier;
022
023import javax.annotation.concurrent.Immutable;
024
025/**
026 * The reasoning behind testing code that doesn't do anything is to achieve 100%
027 * code coverage and to notice when the code
028 * coverage drops. Otherwise one could always think, hey I don't have 100% code
029 * coverage anymore but it's PROBABLY because of my private constructors
030 * (or any other code that's not used but preferred to have).
031 * This makes it easy to spot untested methods without having to check that it
032 * only was a private constructor etc.
033 */
034@Immutable
035public final class UtilityClassTester
036{
037        private UtilityClassTester()
038        {
039        }
040
041        /**
042         * Detects if a utility class isn't final or if its no-args constructor
043         * isn't private. Also calls the constructor to get code coverage for it.
044         * The constructor is allowed to throw exceptions.
045         */
046        public static void testUtilityClassDesign(Class<?> ... utilityClasses)
047        {
048                for(Class<?> clazz : utilityClasses)
049                {
050                        try
051                        {
052                                Constructor<?> constructor = clazz.getDeclaredConstructor();
053                                assertThat(constructor.getModifiers() & Modifier.PRIVATE).as("Constructor for " + clazz + " should be private.")
054                                                .isEqualTo(Modifier.PRIVATE);
055                                constructor.setAccessible(true);
056                                assertThat(constructor.newInstance()).isNotNull();
057                                assertThat(clazz.getModifiers() & Modifier.FINAL).as("Utility class " + clazz + " not final").isEqualTo(Modifier.FINAL);
058                        }
059                        catch(NoSuchMethodException e)
060                        {
061                                throw new IllegalArgumentException("No no-arg constructor found for: " + clazz.getName(), e);
062                        }
063                        catch(InvocationTargetException e)
064                        {
065                                // Some utility classes like to throw from their private constructor just to be on
066                                // the safe side, let's accept that
067                        }
068                        catch(InstantiationException e)
069                        {
070                                throw new IllegalArgumentException("Utility class " + clazz.getName() + " may not be abstract", e);
071                        }
072                        catch(IllegalAccessException e)
073                        {
074                                // Let's just say that if we're not able to create an instance it's fine, after all
075                                // that's what we're testing here
076                        }
077                }
078        }
079}
