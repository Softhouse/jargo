015    package se.softhouse.common.testlib;
016    
017    import org.fest.assertions.Fail;
018    
019    /**
020     * Used to simulate an exception. Commonly used to avoid throwing something like
021     * {@link IllegalStateException} or {@link IllegalArgumentException} in tests (catching those could
022     * mask other errors if the message on the exception isn't verified, and asserting that message
023     * would only cause further confusion as that's not really what the test is about). Catching
024     * {@link SimulatedException} should be safe in tests as that's the purpose of the test, to catch it
025     * and verify behavior when it occurs. <br>
026     * <b>Reminder:</b> don't forget to call {@link Fail#fail(String) fail} if the exception doesn't
027     * happen (otherwise you won't notice any test failures and a kitten may die)
028     */
029    public class SimulatedException extends RuntimeException
030    {
031            private static final long serialVersionUID = 1L;
032    
033            /**
034             * @see RuntimeException#RuntimeException()
035             */
036            public SimulatedException()
037            {
038            }
039    
040            /**
041             * @see RuntimeException#RuntimeException(String)
042             */
043            public SimulatedException(String message)
044            {
045                    super(message);
046            }
047    }
