015    package se.softhouse.common.testlib;
016    
017    import java.io.InputStream;
018    import java.io.InputStreamReader;
019    import java.util.concurrent.Callable;
020    import java.util.concurrent.Executors;
021    import java.util.concurrent.Future;
022    
023    import javax.annotation.concurrent.Immutable;
024    
025    import com.google.common.base.Charsets;
026    import com.google.common.io.CharStreams;
027    
028    /**
029     * Utility for working with {@link Readable} and {@link InputStream stream} instances.
030     */
031    @Immutable
032    public final class Streams
033    {
034            private Streams()
035            {
036            }
037    
038            /**
039             * Reads from {@code source} asynchronously. Use {@link Future#get()} to get the
040             * characters read. Characters will be read using {@link Charsets#UTF_8 UTF-8}. A typical use
041             * case is to avoid <a
042             * href="http://www.javaworld.com/javaworld/jw-12-2000/jw-1229-traps.html?page=4">deadlock
043             * pitfalls</a> when launching {@link Process processes}.
044             */
045            public static Future<String> readAsynchronously(InputStream source)
046            {
047                    final Readable sourceInUTF8 = new InputStreamReader(source, Charsets.UTF_8);
048                    Callable<String> reader = new Callable<String>(){
049                            @Override
050                            public String call() throws Exception
051                            {
052                                    StringBuilder result = new StringBuilder();
053                                    CharStreams.copy(sourceInUTF8, result);
054                                    return result.toString();
055                            }
056                    };
057                    return Executors.newSingleThreadExecutor().submit(reader);
058            }
059    }
