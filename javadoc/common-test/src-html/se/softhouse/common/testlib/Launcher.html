015package se.softhouse.common.testlib;
016
017import static com.google.common.base.Preconditions.checkArgument;
018import static com.google.common.base.Preconditions.checkNotNull;
019import static com.google.common.io.ByteStreams.toByteArray;
020import static java.lang.reflect.Modifier.isPublic;
021import static java.lang.reflect.Modifier.isStatic;
022
023import java.io.File;
024import java.io.IOException;
025import java.util.Arrays;
026import java.util.List;
027
028import com.google.common.base.Charsets;
029import com.google.common.collect.Lists;
030
031/**
032 * Can launch java programs for a {@link Class} with a main method. Output is captured in the
033 * returned {@link LaunchedProgram}.
034 */
035public final class Launcher
036{
037        /**
038         * Result from a {@link Launcher#launch(Class, String...) launched program}
039         */
040        public static final class LaunchedProgram
041        {
042                private final String errors;
043                private final String output;
044                private final String jvm;
045                private final String classPath;
046
047                private LaunchedProgram(String errors, String output, String jvm, String classPath)
048                {
049                        this.errors = errors;
050                        this.output = output;
051                        this.classPath = classPath;
052                        this.jvm = jvm;
053                }
054
055                /**
056                 * The {@link System#out} in {@link Charsets#UTF_8 UTF-8} from the launched program
057                 */
058                public String output()
059                {
060                        return output;
061                }
062
063                /**
064                 * The {@link System#err} in {@link Charsets#UTF_8 UTF-8} from the launched program
065                 */
066                public String errors()
067                {
068                        return errors;
069                }
070
071                /**
072                 * Returns information suitable to print in case of errors on a debug level
073                 */
074                public String debugInformation()
075                {
076                        return "\njvm: " + jvm + "\nclasspath: " + classPath;
077                }
078        }
079
080        private Launcher()
081        {
082        }
083
084        /**
085         * Runs {@code classWithMainMethod} in a separate process using the system property
086         * {@code java.home} to find java and {@code java.class.path} for the class path. This method
087         * will wait until program execution has finished.
088         * 
089         * @param classWithMainMethod a class with a static "main" method
090         * @param programArguments optional arguments to pass to the program
091         * @return output/errors from the executed program
092         * @throws IOException if an I/O error occurs while starting {@code classWithMainMethod} as a
093         *             process
094         * @throws InterruptedException if the thread starting the program is
095         *             {@link Thread#interrupted()}
096         * @throws IllegalArgumentException if {@code classWithMainMethod} doesn't have a static main
097         *             method
098         * @throws SecurityException if it's not possible to validate the existence of a main method in
099         *             {@code classWithMainMethod}
100         */
101        public static LaunchedProgram launch(Class<?> classWithMainMethod, String ... programArguments) throws IOException, InterruptedException
102        {
103                checkNotNull(classWithMainMethod);
104                checkNotNull(programArguments);
105                try
106                {
107
108                        int modifiers = classWithMainMethod.getDeclaredMethod("main", String[].class).getModifiers();
109                        boolean validModifiers = isStatic(modifiers) && isPublic(modifiers);
110                        checkArgument(validModifiers, "%s's main method needs to be static and public for it to be launchable", classWithMainMethod.getName());
111                }
112                catch(NoSuchMethodException e)
113                {
114                        throw new IllegalArgumentException("No main method found on: " + classWithMainMethod.getName(), e);
115                }
116                String jvm = new File(new File(System.getProperty("java.home"), "bin"), "java").getAbsolutePath();
117                String classPath = System.getProperty("java.class.path");
118
119                List<String> args = Lists.newArrayList(jvm, "-cp", classPath, classWithMainMethod.getName());
120                args.addAll(Arrays.asList(programArguments));
121
122                Process program = new ProcessBuilder().command(args).start();
123                String output = new String(toByteArray(program.getInputStream()), Charsets.UTF_8);
124                String errors = new String(toByteArray(program.getErrorStream()), Charsets.UTF_8);
125
126                program.waitFor();
127                return new LaunchedProgram(errors, output, jvm, classPath);
128        }
129}
