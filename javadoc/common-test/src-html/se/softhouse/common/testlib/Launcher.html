015package se.softhouse.common.testlib;
016
017import static com.google.common.base.Preconditions.checkArgument;
018import static com.google.common.base.Preconditions.checkNotNull;
019import static com.google.common.io.ByteStreams.toByteArray;
020import static java.lang.reflect.Modifier.isStatic;
021
022import java.io.File;
023import java.io.IOException;
024import java.util.Arrays;
025import java.util.List;
026
027import com.google.common.base.Charsets;
028import com.google.common.collect.Lists;
029
030public final class Launcher
031{
032        public static class LaunchedProgram
033        {
034                /**
035                 * The {@link System#err} in {@link Charsets#UTF_8 UTF-8} from the launched program
036                 */
037                public String errors;
038                /**
039                 * The {@link System#out} in {@link Charsets#UTF_8 UTF-8} from the launched program
040                 */
041                public String output;
042                private final String jvm;
043                private final String classPath;
044
045                private LaunchedProgram(String errors, String output, String jvm, String classPath)
046                {
047                        this.errors = errors;
048                        this.output = output;
049                        this.classPath = classPath;
050                        this.jvm = jvm;
051                }
052
053                public String debugInformation()
054                {
055                        return "\njvm: " + jvm + "\nclasspath: " + classPath;
056                }
057        }
058
059        private Launcher()
060        {
061        }
062
063        /**
064         * Runs {@code classWithMainMethod} in a separate process using the system property
065         * {@code java.home} to find java and {@code java.class.path} for the class path. This method
066         * will wait until program execution has finished.
067         * 
068         * @param classWithMainMethod a class with a static "main" method
069         * @param programArguments optional arguments to pass to the program
070         * @return output/errors from the executed program
071         * @throws IOException if an I/O error occurs while starting {@code classWithMainMethod} as a
072         *             process
073         * @throws InterruptedException if the thread starting the program is
074         *             {@link Thread#interrupted()}
075         * @throws IllegalArgumentException if {@code classWithMainMethod} doesn't have a static main
076         *             method
077         * @throws SecurityException if it's not possible to validate the existence of a main method in
078         *             {@code classWithMainMethod}
079         */
080        public static LaunchedProgram launch(Class<?> classWithMainMethod, String ... programArguments) throws IOException, InterruptedException
081        {
082                checkNotNull(classWithMainMethod);
083                checkNotNull(programArguments);
084                try
085                {
086
087                        int modifiers = classWithMainMethod.getDeclaredMethod("main", String[].class).getModifiers();
088                        boolean validModifiers = isStatic(modifiers);// && isPublic(modifiers);
089                        checkArgument(validModifiers, "%s's main method needs to be static and public for it to be launchable", classWithMainMethod.getName());
090                }
091                catch(NoSuchMethodException e)
092                {
093                        throw new IllegalArgumentException("No main method found on: " + classWithMainMethod.getName());
094                }
095                String jvm = new File(new File(System.getProperty("java.home"), "bin"), "java").getAbsolutePath();
096                String classPath = System.getProperty("java.class.path");
097
098                List<String> args = Lists.newArrayList(jvm, "-cp", classPath, classWithMainMethod.getName());
099                args.addAll(Arrays.asList(programArguments));
100
101                Process program = new ProcessBuilder().command(args).start();
102                String output = new String(toByteArray(program.getInputStream()), Charsets.UTF_8);
103                String errors = new String(toByteArray(program.getErrorStream()), Charsets.UTF_8);
104
105                program.waitFor();
106                return new LaunchedProgram(errors, output, jvm, classPath);
107        }
108}
