015    package se.softhouse.common.testlib;
016    
017    import static com.google.common.base.Preconditions.checkNotNull;
018    
019    import java.io.ByteArrayInputStream;
020    import java.io.ByteArrayOutputStream;
021    import java.io.IOException;
022    import java.io.ObjectInputStream;
023    import java.io.ObjectOutputStream;
024    import java.io.Serializable;
025    
026    import javax.annotation.concurrent.Immutable;
027    
028    /**
029     * Aids tests that needs to test how APIs behave when their objects are serialized
030     */
031    @Immutable
032    public final class Serializer
033    {
034            private Serializer()
035            {
036            }
037    
038            /**
039             * Serializes {@code objectToSerialize} and reads the serialized bytes into an object again.
040             * Requires that readReplace() isn't overridden to return a type not assignable as a {@code T},
041             * for default implementations of {@link Serializable} this is true.
042             */
043            public static <T extends Serializable> T clone(T objectToSerialize)
044            {
045                    checkNotNull(objectToSerialize);
046                    try
047                    {
048                            ByteArrayOutputStream baos = new ByteArrayOutputStream();
049                            ObjectOutputStream oos = new ObjectOutputStream(baos);
050                            oos.writeObject(objectToSerialize);
051                            ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
052                            ObjectInputStream ois = new ObjectInputStream(bais);
053    
054                            @SuppressWarnings("unchecked")
055                            T deserializedObject = (T) ois.readObject();
056                            return deserializedObject;
057                    }
058                    catch(IOException e)
059                    {
060                            throw new IllegalArgumentException("Failed to clone: " + objectToSerialize, e);
061                    }
062                    catch(ClassNotFoundException e)
063                    {
064                            throw new IllegalArgumentException("Failed to clone: " + objectToSerialize, e);
065                    }
066            }
067    }
